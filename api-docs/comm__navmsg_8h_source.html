102  void SetUniqueNumber(uint32_t _UniqueNumber) {
103  value.UnicNumberAndManCode =
104  (value.UnicNumberAndManCode & 0xffe00000) | (_UniqueNumber & 0x1fffff);
105  }
106  void SetManufacturerCode(uint16_t _ManufacturerCode) {
107  value.UnicNumberAndManCode =
108  (value.UnicNumberAndManCode & 0x1fffff) |
109  (((unsigned long)(_ManufacturerCode & 0x7ff)) << 21);
110  }
111  void SetDeviceInstance(unsigned char _DeviceInstance) {
112  value.DeviceInstance = _DeviceInstance;
113  }
114  void SetDeviceFunction(unsigned char _DeviceFunction) {
115  value.DeviceFunction = _DeviceFunction;
116  }
117  void SetDeviceClass(unsigned char _DeviceClass) {
118  value.DeviceClass = ((_DeviceClass & 0x7f) << 1);
119  }
120  void SetIndustryGroup(unsigned char _IndustryGroup) {
121  value.IndustryGroupAndSystemInstance =
122  (value.IndustryGroupAndSystemInstance & 0x0f) | (_IndustryGroup << 4) |
123  0x80;
124  }
125  void SetSystemInstance(unsigned char _SystemInstance) {
126  value.IndustryGroupAndSystemInstance =
127  (value.IndustryGroupAndSystemInstance & 0xf0) |
128  (_SystemInstance & 0x0f);
