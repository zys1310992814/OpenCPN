185 // https://wiki.openstreetmap.org/wiki/Slippy_map_tilenames#C.2FC.2B.2B
186 static int long2tilex(double lon, int z) {
187  if (lon < -180) lon += 360;
188 
189  return (int)(floor((lon + 180.0) / 360.0 * pow(2.0, z)));
190 }
191 
192 static int lat2tiley(double lat, int z) {
193  int y = (int)(floor(
194  (1.0 -
195  log(tan(lat * M_PI / 180.0) + 1.0 / cos(lat * M_PI / 180.0)) / M_PI) /
196  2.0 * pow(2.0, z)));
197  int ymax = 1 << z;
198  y = ymax - y - 1;
199  return y;
200 }
201 
202 static double tilex2long(int x, int z) { return x / pow(2.0, z) * 360.0 - 180; }
203 
204 static double tiley2lat(int y, int z) {
205  double n = pow(2.0, z);
206  int ymax = 1 << z;
207  y = ymax - y - 1;
208  double latRad = atan(sinh(M_PI * (1 - (2 * y / n))));
209  return 180.0 / M_PI * latRad;
210 }

335 // Checks/corrects/completes the initialization based on real data from the
336 // tiles table
337 void ChartMBTiles::InitFromTiles(const wxString &name) {
338  try {
339  // Open the MBTiles database file
340  const char *name_UTF8 = "";
341  wxCharBuffer utf8CB = name.ToUTF8(); // the UTF-8 buffer
342  if (utf8CB.data()) name_UTF8 = utf8CB.data();
343 
344  SQLite::Database db(name_UTF8);
345 
346  // Check if tiles with advertised min and max zoom level really exist, or
347  // correct the defaults We can't blindly use what we find though - the DB
348  // often contains empty cells at very low zoom levels, so if we have some
349  // info from metadata, we will use that if more conservative...
350  SQLite::Statement query(db,
351  "SELECT min(zoom_level) AS min_zoom, "
352  "max(zoom_level) AS max_zoom FROM tiles");
353  while (query.executeStep()) {
354  const char *colMinZoom = query.getColumn(0);
355  const char *colMaxZoom = query.getColumn(1);
356 
357  int min_zoom = 0, max_zoom = 0;
358  sscanf(colMinZoom, "%i", &min_zoom);
359  m_minZoom = wxMax(m_minZoom, min_zoom);
360  sscanf(colMaxZoom, "%i", &max_zoom);
361  m_maxZoom = wxMin(m_maxZoom, max_zoom);
362  if (m_minZoom > m_maxZoom) {
363  // We are looking at total nonsense with wrong metatadata and actual
364  // tile coverage out of it, better use what's really in the data to be
365  // able to show at least something
366  m_minZoom = min_zoom;
367  m_maxZoom = max_zoom;
368  }
369  }
370 
371  // std::cout << name.c_str() << " zoom_min: " << m_minZoom << "
372  // zoom_max: " << m_maxZoom << std::endl;
373 
374  // Traversing the entire tile table can be expensive....
375  // Use declared bounds if present.
376 
377  if (!std::isnan(m_LatMin) && !std::isnan(m_LatMax) &&
378  !std::isnan(m_LonMin) && !std::isnan(m_LonMax))
379  return;
380 
381  // Try to guess the coverage extents from the tiles. This will be hard to
382  // get right -
383  // the finest resolution likely does not cover the whole area, while the
384  // lowest resolution tiles probably contain a lot of theoretical space which
385  // actually is not covered. And some resolutions may be actually missing...
386  // What do we use?
387  // If we have the metadata and it is not completely off, we should probably
388  // prefer it.
389  SQLite::Statement query1(
390  db,
391  wxString::Format(
392  "SELECT min(tile_row) AS min_row, max(tile_row) as max_row, "
393  "min(tile_column) as min_column, max(tile_column) as max_column, "
394  "count(*) as cnt, zoom_level FROM tiles WHERE zoom_level >= %d "
395  "AND zoom_level <= %d GROUP BY zoom_level ORDER BY zoom_level ASC",
396  m_minZoom, m_maxZoom)
397  .c_str());
398  float minLat = 999., maxLat = -999.0, minLon = 999., maxLon = -999.0;
399  while (query1.executeStep()) {
400  const char *colMinRow = query1.getColumn(0);
401  const char *colMaxRow = query1.getColumn(1);
402  const char *colMinCol = query1.getColumn(2);
403  const char *colMaxCol = query1.getColumn(3);
404  const char *colCnt = query1.getColumn(4);
405  const char *colZoom = query1.getColumn(5);
406 
407  int minRow, maxRow, minCol, maxCol, cnt, zoom;
408  sscanf(colMinRow, "%i", &minRow);
409  sscanf(colMaxRow, "%i", &maxRow);
410  sscanf(colMinCol, "%i", &minCol);
411  sscanf(colMaxCol, "%i", &maxCol);
412  sscanf(colMinRow, "%i", &minRow);
413  sscanf(colMaxRow, "%i", &maxRow);
414  sscanf(colCnt, "%i", &cnt);
415  sscanf(colZoom, "%i", &zoom);
416 
417  // Let's try to use the simplest possible algo and just look for the zoom
418  // level with largest extent (Which probably be the one with lowest
419  // resolution?)...
420  minLat = wxMin(minLat, tiley2lat(minRow, zoom));
421  maxLat = wxMax(maxLat, tiley2lat(maxRow - 1, zoom));
422  minLon = wxMin(minLon, tilex2long(minCol, zoom));
423  maxLon = wxMax(maxLon, tilex2long(maxCol + 1, zoom));
424  // std::cout << "Zoom: " << zoom << " minlat: " << tiley2lat(minRow, zoom)
425  // << " maxlat: " << tiley2lat(maxRow - 1, zoom) << " minlon: " <<
426  // tilex2long(minCol, zoom) << " maxlon: " << tilex2long(maxCol + 1, zoom)
427  // << std::endl;
428  }
429 
430  // ... and use what we found only in case we miss some of the values from
431  // metadata...
432  if (std::isnan(m_LatMin)) m_LatMin = minLat;
433  if (std::isnan(m_LatMax)) m_LatMax = maxLat;
434  if (std::isnan(m_LonMin)) m_LonMin = minLon;
435  if (std::isnan(m_LonMax)) m_LonMax = maxLon;
436  } catch (std::exception &e) {
437  const char *t = e.what();
438  wxLogMessage("mbtiles exception: %s", e.what());
439  }
440 }

442 InitReturn ChartMBTiles::Init(const wxString &name, ChartInitFlag init_flags) {
443  m_global_color_scheme = GLOBAL_COLOR_SCHEME_RGB;
444 
445  m_FullPath = name;
446  m_Description = m_FullPath;
447 
448  try {
449  // Open the MBTiles database file
450  const char *name_UTF8 = "";
451  wxCharBuffer utf8CB = name.ToUTF8(); // the UTF-8 buffer
452  if (utf8CB.data()) name_UTF8 = utf8CB.data();
453 
454  SQLite::Database db(name_UTF8);
455 
456  // Compile a SQL query, getting everything from the "metadata" table
457  SQLite::Statement query(db, "SELECT * FROM metadata ");
458 
459  // Loop to execute the query step by step, to get rows of result
460  while (query.executeStep()) {
461  const char *colName = query.getColumn(0);
462  const char *colValue = query.getColumn(1);
463 
464  // Get the geometric extent of the data
465  if (!strncmp(colName, "bounds", 6)) {
466  float lon1, lat1, lon2, lat2;
467  sscanf(colValue, "%g,%g,%g,%g", &lon1, &lat1, &lon2, &lat2);
468 
469  // There is some confusion over the layout of this field...
470  m_LatMax = wxMax(lat1, lat2);
471  m_LatMin = wxMin(lat1, lat2);
472  m_LonMax = wxMax(lon1, lon2);
473  m_LonMin = wxMin(lon1, lon2);
474 
475  }
476 
477 
478  else if(!strncmp(colName, "format", 6) ){
479  m_format = std::string(colValue);
480  }
481 
482  // Get the min and max zoom values present in the db
483  else if (!strncmp(colName, "minzoom", 7)) {
484  sscanf(colValue, "%i", &m_minZoom);
485  } else if (!strncmp(colName, "maxzoom", 7)) {
486  sscanf(colValue, "%i", &m_maxZoom);
487  }
488 
489  else if (!strncmp(colName, "description", 11)) {
490  m_Description = wxString(colValue, wxConvUTF8);
491  } else if (!strncmp(colName, "name", 11)) {
492  m_Name = wxString(colValue, wxConvUTF8);
493  } else if (!strncmp(colName, "type", 11)) {
494  m_Type = wxString(colValue, wxConvUTF8).Upper().IsSameAs("OVERLAY")
495  ? MBTilesType::OVERLAY
496  : MBTilesType::BASE;
497  } else if (!strncmp(colName, "scheme", 11)) {
498  m_Scheme = wxString(colValue, wxConvUTF8).Upper().IsSameAs("XYZ")
499  ? MBTilesScheme::XYZ
500  : MBTilesScheme::TMS;
501  }
502  }
503  } catch (std::exception &e) {
504  const char *t = e.what();
505  wxLogMessage("mbtiles exception: %s", e.what());
506  return INIT_FAIL_REMOVE;
507  }
508 
509  // Fix the missing/wrong metadata values
510  InitFromTiles(name);
511 
512  // set the chart scale parameters based on the max zoom factor
513  m_ppm_avg = 1.0 / OSM_zoomMPP[m_minZoom];
514  m_Chart_Scale = OSM_zoomScale[m_maxZoom];
515 
516  PrepareTiles(); // Initialize the tile data structures
517 
518  LLRegion covrRegion;
519 
520  LLBBox extentBox;
521  extentBox.Set(m_LatMin, m_LonMin, m_LatMax, m_LonMax);
522 
523  const char *name_UTF8 = "";
524  wxCharBuffer utf8CB = name.ToUTF8(); // the UTF-8 buffer
525  if (utf8CB.data()) name_UTF8 = utf8CB.data();
526 
527  SQLite::Database db(name_UTF8);
528 
529  int zoomFactor = m_minZoom;
530  int minRegionZoom = -1;
531  bool covr_populated = false;
532 
533  m_nTiles = 0;
534  while ((zoomFactor <= m_maxZoom) && (minRegionZoom < 0)) {
535  LLRegion covrRegionZoom;
536  wxRegion regionZoom;
537  char qrs[100];
538 
539  // Protect against trying to create the exact coverage for the brutal large
540  // scale layers contianing tens of thousand tiles.
541  sprintf(qrs, "select count(*) from tiles where zoom_level = %d ",
542  zoomFactor);
543  SQLite::Statement query_size(db, qrs);
544 
545  if (query_size.executeStep()) {
546  const char *colValue = query_size.getColumn(0);
547  int tile_at_zoom = atoi(colValue);
548  m_nTiles += tile_at_zoom;
549 
550  if (tile_at_zoom > 1000) {
551  zoomFactor++;
552  if (!covr_populated) {
553  covr_populated = true;
554  covrRegion = extentBox;
555  }
556  continue;
557  }
558  }
559 
560  // query the database
561  sprintf(qrs,
562  "select tile_column, tile_row from tiles where zoom_level = %d ",
563  zoomFactor);
564 
565  // Compile a SQL query, getting the specific data
566  SQLite::Statement query(db, qrs);
567  covr_populated = true;
568 
569  while (query.executeStep()) {
570  const char *colValue = query.getColumn(0);
571  const char *c2 = query.getColumn(1);
572  int tile_x_found = atoi(colValue); // tile_x
573  int tile_y_found = atoi(c2); // tile_y
574 
575  regionZoom.Union(tile_x_found, tile_y_found - 1, 1, 1);
576 
577  } // inner while
578 
579  wxRegionIterator upd(regionZoom); // get the rect list
580  double eps_factor = eps * 100; // roughly 1 m
581 
582  while (upd) {
583  wxRect rect = upd.GetRect();
584 
585  double lonmin =
586  round(tilex2long(rect.x, zoomFactor) / eps_factor) * eps_factor;
587  double lonmax =
588  round(tilex2long(rect.x + rect.width, zoomFactor) / eps_factor) *
589  eps_factor;
590  double latmin =
591  round(tiley2lat(rect.y, zoomFactor) / eps_factor) * eps_factor;
592  double latmax =
593  round(tiley2lat(rect.y + rect.height, zoomFactor) / eps_factor) *
594  eps_factor;
595 
596  LLBBox box;
597  box.Set(latmin, lonmin, latmax, lonmax);
598 
599  LLRegion tileRegion(box);
600  // if(i <= 1)
601  covrRegionZoom.Union(tileRegion);
602 
603  upd++;
604  minRegionZoom = zoomFactor; // We take the first populated (lowest) zoom
605  // level region as the final chart region
606  }
607 
608  covrRegion.Union(covrRegionZoom);
609 
610  zoomFactor++;
611 
612  } // while
613 
614  // The coverage region must be reduced if necessary to include only the db
615  // specified bounds.
616  covrRegion.Intersect(extentBox);
617 
618  m_minZoomRegion = covrRegion;
619 
620  // Populate M_COVR entries for the OCPN chart database
621  if (covrRegion.contours.size()) { // Check for no intersection caused by ??
622  m_nCOVREntries = covrRegion.contours.size();
623  m_pCOVRTablePoints = (int *)malloc(m_nCOVREntries * sizeof(int));
624  m_pCOVRTable = (float **)malloc(m_nCOVREntries * sizeof(float *));
625  std::list<poly_contour>::iterator it = covrRegion.contours.begin();

760 bool ChartMBTiles::getTileTexture(mbTileDescriptor *tile) {
761  if (!m_pDB) return false;
762 
763  // Is the texture ready?
764  if (tile->glTextureName > 0) {
765  glBindTexture(GL_TEXTURE_2D, tile->glTextureName);
766 
767  return true;
768  } else {
769  if (!tile->m_bAvailable) return false;
770  // fetch the tile data from the mbtile database
771  try {
772  char qrs[2100];
773  sprintf(qrs,
774  "select tile_data, length(tile_data) from tiles where zoom_level "
775  "= %d AND tile_column=%d AND tile_row=%d",
776  tile->m_zoomLevel, tile->tile_x, tile->tile_y);
777 
778  // Compile a SQL query, getting the specific blob
779  SQLite::Statement query(*m_pDB, qrs);
780 
781  int queryResult = query.tryExecuteStep();
782  if (SQLITE_DONE == queryResult) {
783  tile->m_bAvailable = false;
784  return false; // requested ROW not found, should never happen
785  } else {
786  SQLite::Column blobColumn = query.getColumn(0); // Get the blob
787  const void *blob = blobColumn.getBlob();
788 
789  int length = query.getColumn(1); // Get the length
790 
791  wxMemoryInputStream blobStream(blob, length);
792  wxImage blobImage;
793 
794  blobImage = wxImage(blobStream, wxBITMAP_TYPE_ANY);
795  int blobWidth, blobHeight;
796  unsigned char *imgdata;
797 
798  if (blobImage.IsOk()){
799  blobWidth = blobImage.GetWidth();
800  blobHeight = blobImage.GetHeight();
801  // Support MapTiler HiDPI tiles, 512x512
802  if ((blobWidth != 256) || (blobHeight != 256))
803  blobImage.Rescale(256, 256, wxIMAGE_QUALITY_NORMAL);
804  imgdata = blobImage.GetData();
805  }
806  else
807  return false;
808 
809  if ((m_global_color_scheme != GLOBAL_COLOR_SCHEME_RGB) &&
810  (m_global_color_scheme != GLOBAL_COLOR_SCHEME_DAY)) {
811  double dimLevel;
812  switch (m_global_color_scheme) {
813  case GLOBAL_COLOR_SCHEME_DUSK: {
814  dimLevel = 0.8;
815  break;
816  }
817  case GLOBAL_COLOR_SCHEME_NIGHT: {
818  dimLevel = 0.3;
819  break;
820  }
821  default: {
822  dimLevel = 1.0;
823  break;
824  }
825  }
826 
827  // for( int iy = 0; iy < blobHeight; iy++ ) {
828  // for( int ix = 0; ix < blobWidth; ix++ ) {
829  // wxImage::RGBValue rgb(
830  // blobImage.GetRed( ix, iy ),
831  // blobImage.GetGreen( ix, iy ),
832  // blobImage.GetBlue( ix, iy ) );
833  // wxImage::HSVValue hsv =
834  // wxImage::RGBtoHSV( rgb );
835  // hsv.value = hsv.value * dimLevel;
836  // wxImage::RGBValue nrgb =
837  // wxImage::HSVtoRGB( hsv );
838  // blobImage.SetRGB( ix, iy,
839  // nrgb.red, nrgb.green, nrgb.blue );
840  // }
841  // }
842 
843  for (int j = 0; j < blobHeight * blobWidth; j++) {
844  unsigned char *d = &imgdata[3 * j];
845  wxImage::RGBValue rgb(*d, *(d + 1), *(d + 2));
846  wxImage::HSVValue hsv = wxImage::RGBtoHSV(rgb);
847  hsv.value = hsv.value * dimLevel;
848  wxImage::RGBValue nrgb = wxImage::HSVtoRGB(hsv);
849  *d = nrgb.red;
850  *(d + 1) = nrgb.green;
851  *(d + 2) = nrgb.blue;
852  }
853  }
854 
855  int stride = 4;
856  int tex_w = 256;
857  int tex_h = 256;
858  if (!imgdata) return false;
859 
860  unsigned char *teximage =
861  (unsigned char *)malloc(stride * tex_w * tex_h);
862  bool transparent = blobImage.HasAlpha();
863 
864  for (int j = 0; j < tex_w * tex_h; j++) {
865  for (int k = 0; k < 3; k++)
866  teximage[j * stride + k] = imgdata[3 * j + k];
867 
868  // Some NOAA Tilesets do not give transparent tiles, so we detect
869  // NOAA's idea of blank as RGB(1,0,0) and force alpha = 0;
870  if (imgdata[3 * j] == 1 && imgdata[3 * j + 1] == 0 &&
871  imgdata[3 * j + 2] == 0) {
872  teximage[j * stride + 3] = 0;
873  } else {
874  if (transparent) {
875  teximage[j * stride + 3] =
876  blobImage.GetAlpha(j % tex_w, j / tex_w);
877  } else {
878  teximage[j * stride + 3] = 255;
879  }
880  }
881  }
882 
883  glGenTextures(1, &tile->glTextureName);
884  glBindTexture(GL_TEXTURE_2D, tile->glTextureName);
885 
886  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
887  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
888  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
889  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
890 
891  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, tex_w, tex_h, 0, GL_RGBA,
892  GL_UNSIGNED_BYTE, teximage);
893 
894  free(teximage);
895 
896  return true;
897  }
898 
899  } catch (std::exception &e) {
900  const char *t = e.what();
901  wxLogMessage("mbtiles exception: %s", e.what());
902  }
903  }
904 
905  return false;
906 }

910 wxPoint2DDouble ChartMBTiles::GetDoublePixFromLL(ViewPort &vp, double lat,
911  double lon) {
912  double easting = 0;
913  double northing = 0;
914  double xlon = lon - eps;
915 
916  switch (vp.m_projection_type) {
917  case PROJECTION_MERCATOR:
918  case PROJECTION_WEB_MERCATOR:
919  default:
920  const double z = WGS84_semimajor_axis_meters * mercator_k0;
921 
922  easting = (xlon - vp.clon) * DEGREE * z;
923 
924  // y =.5 ln( (1 + sin t) / (1 - sin t) )
925  const double s = sin(lat * DEGREE);
926  const double y3 = (.5 * log((1 + s) / (1 - s))) * z;
927 
928  const double s0 = sin(vp.clat * DEGREE);
929  const double y30 = (.5 * log((1 + s0) / (1 - s0))) * z;
930  northing = y3 - y30;
931 
932  break;
933  }
934 
935  double epix = easting * vp.view_scale_ppm;
936  double npix = northing * vp.view_scale_ppm;
937  double dxr = epix;
938  double dyr = npix;
939 
940  // Apply VP Rotation
941  double angle = vp.rotation;
942 
943  if (angle) {
944  dxr = epix * cos(angle) + npix * sin(angle);
945  dyr = npix * cos(angle) - epix * sin(angle);
946  }
947 
948  // printf(" gdpll: %g %g %g\n", vp.clon, (vp.pix_width / 2.0 ) + dxr, (
949  // vp.pix_height / 2.0 ) - dyr);
950 
951  return wxPoint2DDouble((vp.pix_width / 2.0) + dxr,
952  (vp.pix_height / 2.0) - dyr);
953 }

1029  co1[6] = coords[4];
