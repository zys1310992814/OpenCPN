180 void CommDriverN2KSocketCanImpl::SetN2K_Name() {
181  // We choose some "benign" values for OCPN socketCan interface
182  node_name.value.Name = 0;
183 
184  m_unique_number = 1;
185  // Build a simple 16 bit hash of g_hostname, to use as unique "serial number"
186  int hash = 0;
187  std::string str(g_hostname.mb_str());
188  int len = str.size();
189  const char* ch = str.data();
190  for (int i = 0; i < len; i++)
191  hash = hash + ((hash) << 5) + *(ch + i) + ((*(ch + i)) << 7);
192  m_unique_number = ((hash) ^ (hash >> 16)) & 0xffff;
193 
194  node_name.SetManufacturerCode(2046);
195  node_name.SetUniqueNumber(m_unique_number);
196  node_name.SetDeviceFunction(130); // Display
197  node_name.SetDeviceClass(120); // Display
198  node_name.SetIndustryGroup(4); // Marine
199  node_name.SetSystemInstance(0);

310 bool CommDriverN2KSocketCanImpl::SendMessage(std::shared_ptr<const NavMsg> msg,
311  std::shared_ptr<const NavAddr> addr) {
312 
313  wxMutexLocker lock(m_TX_mutex);
314 
315  // Verify claimed address is useable
316  if ( m_source_address < 0)
317  return false;
318 
319  if ( m_source_address > 253) // Could not claim...
320  return false;
321 
322  int socket = GetWorker().GetSocket();
323 
324  if (socket < 0)
325  return false;
326 
327  CanFrame frame;
328  memset(&frame, 0, sizeof(frame));
329 
330  auto msg_n2k = std::dynamic_pointer_cast<const Nmea2000Msg>(msg);
331  std::vector<uint8_t> load = msg_n2k->payload;
332 
333  uint64_t _pgn = msg_n2k->PGN.pgn;
334  auto destination_address = std::static_pointer_cast<const NavAddr2000>(addr);
335 
336 
337  unsigned long canId = BuildCanID(msg_n2k->priority, m_source_address,
338  destination_address->address, _pgn);
339 
340  frame.can_id = canId | CAN_EFF_FLAG;
341 
342  if (load.size() <= 8){
343  frame.can_dlc = load.size();
344  if (load.size() > 0)
345  memcpy(&frame.data, load.data(), load.size());
346 
347  int sentbytes = write(socket, &frame, sizeof(frame));
348  }
349  else { // Fast Packet
350  int sequence = (m_last_TX_sequence + 0x20) & 0xE0;
351  m_last_TX_sequence = sequence;
352  unsigned char *data_ptr = load.data();
353  int n_remaining = load.size();
354 
355  // First packet
356  frame.can_dlc = 8;
357  frame.data[0] = sequence;
358  frame.data[1] = load.size();
359  int data_len_0 = wxMin(load.size(), 6);
360  memcpy(&frame.data[2], load.data(), data_len_0);
361 
362  int sentbytes0 = write(socket, &frame, sizeof(frame));
363 
364  data_ptr += data_len_0;
365  n_remaining -= data_len_0;
366  sequence++;
367 
368  // The rest of the bytes
369  while (n_remaining > 0){
370  wxMilliSleep(10);
371  frame.data[0] = sequence;
372  int data_len_n = wxMin(n_remaining, 7);
373  memcpy(&frame.data[1], data_ptr, data_len_n);
374 
375  int sentbytesn = write(socket, &frame, sizeof(frame));
376 
377  data_ptr += data_len_n;
378  n_remaining -= data_len_n;
379  sequence++;
380  }
381  }
382 
383 
384  return true;
385 }

460  data.push_back(fast_messages[position].data[n]);

484 int Worker::InitSocket(const std::string port_name) {
485  int sock = socket(PF_CAN, SOCK_RAW, CAN_RAW);
486  if (sock < 0) {
487  SocketMessage("SocketCAN socket create failed: ", port_name);
488  return -1;
489  }
490 
491  // Get the interface index
492  struct ifreq if_request;
493  strcpy(if_request.ifr_name, port_name.c_str());
494  if (ioctl(sock, SIOCGIFINDEX, &if_request) < 0) {
495  SocketMessage("SocketCAN ioctl (SIOCGIFINDEX) failed: ", port_name);
496  return -1;
497  }
498 
499  // Check if interface is UP
500  struct sockaddr_can can_address;
501  can_address.can_family = AF_CAN;
502  can_address.can_ifindex = if_request.ifr_ifindex;
503  if (ioctl(sock, SIOCGIFFLAGS, &if_request) < 0) {
504  SocketMessage("SocketCAN socket IOCTL (SIOCGIFFLAGS) failed: ", port_name);
505  return -1;
506  }
507  if (if_request.ifr_flags & IFF_UP) {
508  ThreadMessage("socketCan interface is UP");
509  } else {
510  ThreadMessage("socketCan interface is NOT UP");
511  return -1;
512  }
513 
514  // Set timeout and bind
515  struct timeval tv;
516  tv.tv_sec = kSocketTimeoutSeconds;
517  tv.tv_usec = 0;
518  int r =
519  setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, (const char*)&tv, sizeof tv);
520  if (r < 0) {
521  SocketMessage("SocketCAN setsockopt SO_RCVTIMEO failed on device: ",
522  port_name);
523  return -1;
524  }
525  r = bind(sock, (struct sockaddr*)&can_address, sizeof(can_address));
526  if (r < 0) {
527  SocketMessage("SocketCAN socket bind() failed: ", port_name);
528  return -1;
529  }
530  return sock;
531 }

539 void Worker::HandleInput(CanFrame frame) {
540  int position = -1;
541  bool ready = true;
542 
543  CanHeader header(frame);
544  if (header.IsFastMessage()) {
545  position = fast_messages.FindMatchingEntry(header, frame.data[0]);
546  if (position == kNotFound) {
547  // Not an existing fast message: create new entry and insert first frame
548  position = fast_messages.AddNewEntry();
549  ready = fast_messages.InsertEntry(header, frame.data, position);
550  } else {
551  // An existing fast message entry is present, append the frame
552  ready = fast_messages.AppendEntry(header, frame.data, position);
553  }
554  }
555  if (ready) {
556  std::vector<unsigned char> vec;
557  if (position >= 0) {
558  // Re-assembled fast message
559  vec = PushFastMsgFragment(header, position);
560  } else {
561  // Single frame message
562  vec = PushCompleteMsg(header, position, frame);
563  }
564  //auto name = N2kName(static_cast<uint64_t>(header.pgn));
565  auto src_addr = m_parent_driver->GetAddress(m_parent_driver->node_name);
566  auto msg = std::make_shared<const Nmea2000Msg>(header.pgn, vec, src_addr);
567  auto msg_all = std::make_shared<const Nmea2000Msg>(1, vec, src_addr);
568 
569  ProcessRxMessages(msg);
570  m_parent_driver->m_listener.Notify(std::move(msg));
571  m_parent_driver->m_listener.Notify(std::move(msg_all));
572 
573  }

577 void Worker::ProcessRxMessages(std::shared_ptr<const Nmea2000Msg> n2k_msg){
578 
579  if(n2k_msg->PGN.pgn == 59904 ){
580  unsigned long RequestedPGN = 0;
581  RequestedPGN = n2k_msg->payload.at(15) << 16;
582  RequestedPGN += n2k_msg->payload.at(14) << 8;
583  RequestedPGN += n2k_msg->payload.at(13);
584 
585  switch (RequestedPGN){
586  case 60928:
587  m_parent_driver->SendAddressClaim(m_parent_driver->m_source_address);
588  break;
589  case 126996:
590  m_parent_driver->SendProductInfo();
591  break;
592  default:
593  break;
594  }
595  }
596 
597  else if(n2k_msg->PGN.pgn == 60928 ){
598  // Watch for conflicting source address
599  if (n2k_msg->payload.at(7) == m_parent_driver->m_source_address){
600  // My name
601  uint64_t my_name = m_parent_driver->node_name.GetName();
602 
603  // His name
604  uint64_t his_name = 0;
605  unsigned char *p = (unsigned char *)&his_name;
606  for (unsigned int i=0 ; i < 8 ; i++)
607  *p++ = n2k_msg->payload.at(13 + i);
608 
609  // Compare literally the NAME values
610  if (his_name < my_name){
611  // I lose, so select a new address
612  m_parent_driver->m_source_address++;
613  if ( m_parent_driver->m_source_address > 253)
614  // Could not claim an address
615  m_parent_driver->m_source_address = 254;
616  m_parent_driver->UpdateAttrCanAddress();
617  }
618 
619  // Claim the existing or modified address
620  m_parent_driver->SendAddressClaim(m_parent_driver->m_source_address);
621  }
622  }
623 }

627 void Worker::Entry() {
628  int recvbytes;
629  int socket;
630  CanFrame frame;
631 
632  socket = InitSocket(m_port_name.ToStdString());
633  if (socket < 0) {
634  std::string msg("SocketCAN socket create failed: ");
635  ThreadMessage(msg + m_port_name.ToStdString());
636  m_run_flag = -1;
637  return;
638  }
639  m_socket = socket;
640 
641  // Claim our default address
642  if (m_parent_driver->SendAddressClaim(DEFAULT_N2K_SOURCE_ADDRESS)){
643  m_parent_driver->m_source_address = DEFAULT_N2K_SOURCE_ADDRESS;
644  m_parent_driver->UpdateAttrCanAddress();
645  }
646 
647 
648  // The main loop
649  while (m_run_flag > 0) {
650  recvbytes = read(socket, &frame, sizeof(frame));
651  if (recvbytes == -1) {
652  if (errno == EAGAIN || errno == EWOULDBLOCK) continue; // timeout
653 
654  wxLogWarning("can socket %s: fatal error %s", m_port_name.c_str(),
655  strerror(errno));
656  break;
657  }
658  if (recvbytes != 16) {
659  wxLogWarning("can socket %s: bad frame size: %d (ignored)",
660  m_port_name.c_str(), recvbytes);
661  sleep(1);
662  continue;
663  }
664  HandleInput(frame);
665  }
666  m_run_flag = -1;
667  return;
668 }
