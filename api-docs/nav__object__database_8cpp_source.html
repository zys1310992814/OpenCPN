43 RoutePoint *GPXLoadWaypoint1(pugi::xml_node &wpt_node,
44  wxString def_symbol_name, wxString GUID,
45  bool b_fullviz, bool b_layer,
46  bool b_layerviz, int layer_id) {
47  bool bviz = false;
48  bool bviz_name = false;
49  bool bauto_name = false;
50  bool bshared = false;
51  bool b_propvizname = false;
52  bool b_propviz = false;
53 
54  wxString SymString = def_symbol_name; // default icon
55  wxString NameString;
56  wxString DescString;
57  wxString TideStation;
58  double plan_speed = 0.0;
59  wxString etd;
60  wxString TypeString;
61  wxString GuidString = GUID; // default
62  wxString TimeString;
63  wxDateTime dt;
64  RoutePoint *pWP;
65 
66  HyperlinkList *linklist = NULL;
67 
68  double rlat = wpt_node.attribute("lat").as_double();
69  double rlon = wpt_node.attribute("lon").as_double();
70  double ArrivalRadius = 0;
71  int l_iWaypointRangeRingsNumber = -1;
72  float l_fWaypointRangeRingsStep = -1;
73  int l_pWaypointRangeRingsStepUnits = -1;
74  bool l_bWaypointRangeRingsVisible = false;
75  long l_iWaypointScaleMin = 2147483646;
76  long l_iWaypoinScaleMax = 0;
77  bool l_bWaypointUseScale = false;
78  wxColour l_wxcWaypointRangeRingsColour;
79  l_wxcWaypointRangeRingsColour.Set(_T( "#FFFFFF" ));
80 
81  for (pugi::xml_node child = wpt_node.first_child(); child != 0;
82  child = child.next_sibling()) {
83  const char *pcn = child.name();
84 
85  if (!strcmp(pcn, "sym")) {
86  SymString = wxString::FromUTF8(child.first_child().value());
87  } else if (!strcmp(pcn, "time"))
88  TimeString = wxString::FromUTF8(child.first_child().value());
89 
90  else if (!strcmp(pcn, "name")) {
91  NameString = wxString::FromUTF8(child.first_child().value());
92  if (NameString.StartsWith("@~~")) {
93  // Convert the legacy tidal event definition and change the name so
94  // that it does not kick in next time and cause overiding subsequent
95  // changes
96  TideStation = NameString.Right(NameString.length() - 3);
97  NameString.Replace("@~~", "@-~");
98  }
99  }
100 
101  else if (!strcmp(pcn, "desc")) {
102  DescString = wxString::FromUTF8(child.first_child().value());
103  }
104 
105  else if (!strcmp(pcn, "type")) {
106  TypeString = wxString::FromUTF8(child.first_child().value());
107  }
108 
109  else // Read hyperlink
110  if (!strcmp(pcn, "link")) {
111  wxString HrefString;
112  wxString HrefTextString;
113  wxString HrefTypeString;
114  if (linklist == NULL) linklist = new HyperlinkList;
115  HrefString = wxString::FromUTF8(child.first_attribute().value());
116 
117  for (pugi::xml_node child1 = child.first_child(); child1;
118  child1 = child1.next_sibling()) {
119  wxString LinkString = wxString::FromUTF8(child1.name());
120 
121  if (LinkString == _T ( "text" ))
122  HrefTextString = wxString::FromUTF8(child1.first_child().value());
123  if (LinkString == _T ( "type" ))
124  HrefTypeString = wxString::FromUTF8(child1.first_child().value());
125  }
126 
127  Hyperlink *link = new Hyperlink;
128  link->Link = HrefString;
129  link->DescrText = HrefTextString;
130  link->LType = HrefTypeString;
131  linklist->Append(link);
132  }
133 
134  // OpenCPN Extensions....
135  else if (!strcmp(pcn, "extensions")) {
136  for (pugi::xml_node ext_child = child.first_child(); ext_child;
137  ext_child = ext_child.next_sibling()) {
138  wxString ext_name = wxString::FromUTF8(ext_child.name());
139  if (ext_name == _T ( "opencpn:guid" )) {
140  GuidString = wxString::FromUTF8(ext_child.first_child().value());
141  } else if (ext_name == _T ( "opencpn:viz" )) {
142  b_propviz = true;
143  wxString s = wxString::FromUTF8(ext_child.first_child().value());
144  long v = 0;
145  if (s.ToLong(&v)) bviz = (v != 0);
146  } else if (ext_name == _T ( "opencpn:viz_name" )) {
147  b_propvizname = true;
148  wxString s = wxString::FromUTF8(ext_child.first_child().value());
149  long v = 0;
150  if (s.ToLong(&v)) bviz_name = (v != 0);
151  } else if (ext_name == _T ( "opencpn:auto_name" )) {
152  wxString s = wxString::FromUTF8(ext_child.first_child().value());
153  long v = 0;
154  if (s.ToLong(&v)) bauto_name = (v != 0);
155  } else if (ext_name == _T ( "opencpn:shared" )) {
156  wxString s = wxString::FromUTF8(ext_child.first_child().value());
157  long v = 0;
158  if (s.ToLong(&v)) bshared = (v != 0);
159  }
160  if (ext_name == _T ( "opencpn:arrival_radius" )) {
161  wxString::FromUTF8(ext_child.first_child().value())
162  .ToDouble(&ArrivalRadius);
163  }
164  if (ext_name == _T("opencpn:waypoint_range_rings")) {
165  for (pugi::xml_attribute attr = ext_child.first_attribute(); attr;
166  attr = attr.next_attribute()) {
167  if (wxString::FromUTF8(attr.name()) == _T("number"))
168  l_iWaypointRangeRingsNumber = attr.as_int();
169  else if (wxString::FromUTF8(attr.name()) == _T("step"))
170  l_fWaypointRangeRingsStep = attr.as_float();
171  else if (wxString::FromUTF8(attr.name()) == _T("units"))
172  l_pWaypointRangeRingsStepUnits = attr.as_int();
173  else if (wxString::FromUTF8(attr.name()) == _T("visible"))
174  l_bWaypointRangeRingsVisible = attr.as_bool();
175  else if (wxString::FromUTF8(attr.name()) == _T("colour"))
176  l_wxcWaypointRangeRingsColour.Set(
177  wxString::FromUTF8(attr.as_string()));
178  }
179  }
180  if (ext_name == _T("opencpn:scale_min_max")) {
181  for (pugi::xml_attribute attr = ext_child.first_attribute(); attr;
182  attr = attr.next_attribute()) {
183  if (wxString::FromUTF8(attr.name()) == _T("UseScale"))
184  l_bWaypointUseScale = attr.as_bool();
185  else if (wxString::FromUTF8(attr.name()) == _T("ScaleMin"))
186  l_iWaypointScaleMin = attr.as_int();
187  else if (wxString::FromUTF8(attr.name()) == _T("ScaleMax"))
188  l_iWaypoinScaleMax = attr.as_float();
189  }
190  }
191  if (ext_name == _T ( "opencpn:tidestation" )) {
192  TideStation = wxString::FromUTF8(ext_child.first_child().value());
193  }
194  if (ext_name == _T ( "opencpn:rte_properties" )) {
195  for (pugi::xml_attribute attr = ext_child.first_attribute(); attr;
196  attr = attr.next_attribute()) {
197  if (!strcmp(attr.name(), "planned_speed"))
198  plan_speed = attr.as_double();
199  else if (!strcmp(attr.name(), "etd"))
200  etd = attr.as_string();
201  }
202  }
203  } // for
204  } // extensions
205  } // for
206 
207  // Create waypoint
208 
209  if (b_layer) {
210  if (GuidString.IsEmpty()) GuidString = pWayPointMan->CreateGUID(NULL);
211  }
212 
213  pWP = new RoutePoint(rlat, rlon, SymString, NameString, GuidString,
214  false); // do not add to global WP list yet...
215  pWP->m_MarkDescription = DescString;
216  pWP->m_TideStation = TideStation;
217  pWP->m_bIsolatedMark = bshared; // This is an isolated mark
218  pWP->SetWaypointArrivalRadius(ArrivalRadius);
219  pWP->SetWaypointRangeRingsNumber(l_iWaypointRangeRingsNumber);
220  pWP->SetWaypointRangeRingsStep(l_fWaypointRangeRingsStep);
221  pWP->SetWaypointRangeRingsStepUnits(l_pWaypointRangeRingsStepUnits);
222  pWP->SetShowWaypointRangeRings(l_bWaypointRangeRingsVisible);
223 
224  // Migrate from O4.x XML format.
225  // In O5, the attribute "range rings visible" is synonymous with ( "range
226  // rings number" != 0 ) So, if we see an attribute "visible"=false in
227  // importing from XML, we must set "number" = 0 to be consistent
228  if (!l_bWaypointRangeRingsVisible) pWP->SetWaypointRangeRingsNumber(0);
229 
230  pWP->SetWaypointRangeRingsColour(l_wxcWaypointRangeRingsColour);
231  pWP->SetScaMin(l_iWaypointScaleMin);
232  pWP->SetScaMax(l_iWaypoinScaleMax);
233  pWP->SetUseSca(l_bWaypointUseScale);
234  pWP->SetPlannedSpeed(plan_speed);
235  pWP->SetETD(etd);
236 
237  pWP->m_bShowNameData = bviz_name;
238  if (b_propvizname)
239  pWP->m_bShowName = bviz_name;
240  else if (b_fullviz)
241  pWP->m_bShowName = true;
242  else
243  pWP->m_bShowName = false;
244 
245  if (b_propviz)
246  pWP->m_bIsVisible = bviz;
247  else if (b_fullviz)
248  pWP->m_bIsVisible = true;
249 
250  if (b_layer) {
251  pWP->m_bIsInLayer = true;
252  pWP->m_LayerID = layer_id;
253  pWP->m_bIsVisible = b_layerviz;
254  pWP->SetListed(false);
255  }
256 
257  pWP->SetShared(bshared);
258  pWP->m_bDynamicName = bauto_name;
259 
260  if (TimeString.Len()) {
261  pWP->m_timestring = TimeString;
262  pWP->SetCreateTime(wxInvalidDateTime); // cause deferred timestamp parsing
263  }
264 
265  if (linklist) {
266  delete pWP->m_HyperlinkList; // created in RoutePoint ctor
267  pWP->m_HyperlinkList = linklist;
268  }
269 
270  return pWP;
271 }

309 
310  wxString Name = wxString::FromUTF8(trk_node.name());

312  pTentTrack = new Track();
313  GPXSeg = 0;
314 
315  TrackPoint *pWp = NULL;
316

318  tschild = tschild.next_sibling()) {

320  if (ChildName == _T ( "trkseg" )) {
321  GPXSeg += 1;
322 
323  // Official GPX spec calls for trkseg to have children trkpt
324  for (pugi::xml_node tpchild = tschild.first_child(); tpchild;
325  tpchild = tpchild.next_sibling()) {
326  wxString tpChildName = wxString::FromUTF8(tpchild.name());
327  if (tpChildName == _T("trkpt")) {
328  pWp = ::GPXLoadTrackPoint1(tpchild);
329  if (pWp){
330  pTentTrack->AddPoint(pWp); // defer BBox calculation
331  pWp->m_GPXTrkSegNo = GPXSeg;
332  }
333  }
334  }
335  } else if (ChildName == _T ( "name" ))
336  TrackName = wxString::FromUTF8(tschild.first_child().value());
337  else if (ChildName == _T ( "desc" ))
338  DescString = wxString::FromUTF8(tschild.first_child().value());
339  else
340 
341  if (ChildName == _T ( "link")) {
342  wxString HrefString;
343  wxString HrefTextString;
344  wxString HrefTypeString;
345  if (linklist == NULL) linklist = new HyperlinkList;
346  HrefString = wxString::FromUTF8(tschild.first_attribute().value());
347 
348  for (pugi::xml_node child1 = tschild.first_child(); child1;
349  child1 = child1.next_sibling()) {
350  wxString LinkString = wxString::FromUTF8(child1.name());
351 
352  if (LinkString == _T ( "text" ))
353  HrefTextString = wxString::FromUTF8(child1.first_child().value());
354  if (LinkString == _T ( "type" ))
355  HrefTypeString = wxString::FromUTF8(child1.first_child().value());
356  }
357 
358  Hyperlink *link = new Hyperlink;
359  link->Link = HrefString;
360  link->DescrText = HrefTextString;
361  link->LType = HrefTypeString;
362  linklist->Append(link);
363  }
364 
365  else if (ChildName == _T ( "extensions" )) {
366  for (pugi::xml_node ext_child = tschild.first_child(); ext_child;
367  ext_child = ext_child.next_sibling()) {
368  wxString ext_name = wxString::FromUTF8(ext_child.name());
369  if (ext_name == _T ( "opencpn:start" )) {
370  pTentTrack->m_TrackStartString =
371  wxString::FromUTF8(ext_child.first_child().value());
372  } else if (ext_name == _T ( "opencpn:end" )) {
373  pTentTrack->m_TrackEndString =
374  wxString::FromUTF8(ext_child.first_child().value());
375  }
376 
377  else if (ext_name == _T ( "opencpn:viz" )) {
378  wxString viz = wxString::FromUTF8(ext_child.first_child().value());
379  b_propviz = true;
380  b_viz = (viz == _T("1"));
381  } else if (ext_name == _T ( "opencpn:style" )) {
382  for (pugi::xml_attribute attr = ext_child.first_attribute(); attr;
383  attr = attr.next_attribute()) {
384  if (!strcmp(attr.name(), "style"))
385  pTentTrack->m_style = (wxPenStyle)attr.as_int();
386  else if (!strcmp(attr.name(), "width"))
387  pTentTrack->m_width = attr.as_int();
388  }
389  }
390 
391  else if (ext_name == _T ( "opencpn:guid" )) {
392  pTentTrack->m_GUID =
393  wxString::FromUTF8(ext_child.first_child().value());
394  }
395 
396  else if (ext_name.EndsWith(
397  _T ( "TrackExtension" ))) // Parse GPXX color
398  {
399  for (pugi::xml_node gpxx_child = ext_child.first_child();
400  gpxx_child; gpxx_child = gpxx_child.next_sibling()) {
401  wxString gpxx_name = wxString::FromUTF8(gpxx_child.name());
402  if (gpxx_name.EndsWith(_T ( "DisplayColor" )))
403  pTentTrack->m_Colour =
404  wxString::FromUTF8(gpxx_child.first_child().value());
405  }
406  }
407  } // extensions
408  }
409  }
410 
411  pTentTrack->SetName(TrackName);
412  pTentTrack->m_TrackDescription = DescString;
413 
414  if (b_propviz)
415  pTentTrack->SetVisible(b_viz);
416  else {
417  if (b_fullviz) pTentTrack->SetVisible();
418  }
419 
420  if (b_layer) {
421  pTentTrack->SetVisible(b_layerviz);
422  pTentTrack->m_bIsInLayer = true;
423  pTentTrack->m_LayerID = layer_id;
424  pTentTrack->SetListed(false);
425  }
426 
427  pTentTrack->SetCurrentTrackSeg(GPXSeg);
428  }
429 
430  if (linklist) {
431  delete pTentTrack->m_HyperlinkList; // created in TrackPoint ctor
432  pTentTrack->m_HyperlinkList = linklist;
433  }
434 
435  return pTentTrack;
436 }

438 Route *GPXLoadRoute1(pugi::xml_node &wpt_node, bool b_fullviz,
439  bool b_layer, bool b_layerviz, int layer_id,
440  bool b_change) {
441  wxString RouteName;
442  wxString DescString;
443  bool b_propviz = false;
444  bool b_propSWPviz = false;
445  bool b_viz = true;
446  bool swpViz = false;
447  Route *pTentRoute = NULL;
448 
449  wxString Name = wxString::FromUTF8(wpt_node.name());
450  if (Name == _T ( "rte" )) {
451  pTentRoute = new Route();
452  HyperlinkList *linklist = NULL;
453 
454  RoutePoint *pWp = NULL;
455  bool route_existing = false;
456  pTentRoute->m_TimeDisplayFormat = RTE_TIME_DISP_UTC;
457 
458  for (pugi::xml_node tschild = wpt_node.first_child(); tschild;
459  tschild = tschild.next_sibling()) {
460  wxString ChildName = wxString::FromUTF8(tschild.name());
461 
462  // load extentions first to determine if the route still exists
463  if (ChildName == _T ( "extensions" )) {
464  for (pugi::xml_node ext_child = tschild.first_child(); ext_child;
465  ext_child = ext_child.next_sibling()) {
466  wxString ext_name = wxString::FromUTF8(ext_child.name());
467 
468  if (ext_name == _T ( "opencpn:start" )) {
469  pTentRoute->m_RouteStartString =
470  wxString::FromUTF8(ext_child.first_child().value());
471  } else if (ext_name == _T ( "opencpn:end" )) {
472  pTentRoute->m_RouteEndString =
473  wxString::FromUTF8(ext_child.first_child().value());
474  }
475 
476  else if (ext_name == _T ( "opencpn:viz" )) {
477  wxString viz = wxString::FromUTF8(ext_child.first_child().value());
478  b_propviz = true;
479  b_viz = (viz == _T("1"));
480  }
481 
482  else if (ext_name == _T ( "opencpn:sharedWPviz" )) {
483  wxString viz = wxString::FromUTF8(ext_child.first_child().value());
484  b_propSWPviz = true;
485  swpViz = (viz == _T("1"));
486  } else if (ext_name == _T ( "opencpn:style" )) {
487  for (pugi::xml_attribute attr = ext_child.first_attribute(); attr;
488  attr = attr.next_attribute()) {
489  if (!strcmp(attr.name(), "style"))
490  pTentRoute->m_style = (wxPenStyle)attr.as_int();
491  else if (!strcmp(attr.name(), "width"))
492  pTentRoute->m_width = attr.as_int();
493  }
494  }
495 
496  else if (ext_name == _T ( "opencpn:guid" )) {
497  pTentRoute->m_GUID =
498  wxString::FromUTF8(ext_child.first_child().value());
499  }
500 
501  else if (ext_name == _T ( "opencpn:planned_speed" )) {
502  pTentRoute->m_PlannedSpeed = atof(ext_child.first_child().value());
503  }
504 
505  else if (ext_name == _T ( "opencpn:planned_departure" )) {
506  ParseGPXDateTime(
507  pTentRoute->m_PlannedDeparture,
508  wxString::FromUTF8(ext_child.first_child().value()));
509  }
510 
511  else if (ext_name == _T ( "opencpn:time_display" )) {
512  pTentRoute->m_TimeDisplayFormat =
513  wxString::FromUTF8(ext_child.first_child().value());
514  } else if (ext_name.EndsWith(
515  _T ( "RouteExtension" ))) // Parse GPXX color
516  {
517  for (pugi::xml_node gpxx_child = ext_child.first_child();
518  gpxx_child; gpxx_child = gpxx_child.next_sibling()) {
519  wxString gpxx_name = wxString::FromUTF8(gpxx_child.name());
520  if (gpxx_name.EndsWith(_T ( "DisplayColor" )))
521  pTentRoute->m_Colour =
522  wxString::FromUTF8(gpxx_child.first_child().value());
523  }
524  }
525  }
526  if (!b_change) {
527  if (RouteExists(pTentRoute->m_GUID)) { // we are loading a different
528  // route with the same guid so
529  // let's generate a new guid
530  pTentRoute->m_GUID = pWayPointMan->CreateGUID(NULL);
531  route_existing = true;
532  }
533  }
534  } // extension
535  else if (ChildName == _T ( "rtept" )) {
536  RoutePoint *tpWp =
537  ::GPXLoadWaypoint1(tschild, _T("square"), _T(""), b_fullviz,
538  b_layer, b_layerviz, layer_id);
539  RoutePoint *erp = NULL;
540  if (!b_layer) erp = ::WaypointExists(tpWp->m_GUID);
541  // 1) if b_change is true, that means we are after crash - load the
542  // route and points as found in source file 2) if route_existing, we are
543  // loading a different route with the same guid. In this case load
544  // points as found in
545  // source file, changing the guid, but keep existing "isolated point" as
546  // found in the DB
547  // 3) in all other cases keep existing points if found and load new
548  // points if not found
549  bool new_wpt = true;
550  if (b_change)
551  pWp = tpWp;
552  else {
553  if (erp != NULL &&
554  (!route_existing || (route_existing && tpWp->IsShared()))) {
555  pWp = erp;
556  new_wpt = false;
557  } else {
558  if (route_existing) tpWp->m_GUID = pWayPointMan->CreateGUID(NULL);
559  pWp = tpWp;
560  }
561  }
562 
563  pTentRoute->AddPoint(pWp, false, true); // defer BBox calculation
