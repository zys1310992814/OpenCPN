59 wxString FormatPrintableMessage(wxString msg_raw) {
60  std::string fmsg;
61  std::string str = msg_raw.ToStdString();
62  for (std::string::iterator it = str.begin(); it != str.end(); ++it) {
63  if (isprint(*it))
64  fmsg += *it;
65  else {
66  wxString bin_print;
67  bin_print.Printf("<0x%02X>", *it);
68  fmsg += bin_print;
69  }
70  }
71 
72  return wxString(fmsg.c_str());
73 }

95 void BroadcastNMEA0183Message(const wxString& msg, NmeaLog& nmea_log,
96  EventVar& on_msg_sent) {
97  auto& registry = CommDriverRegistry::GetInstance();
98  const std::vector<std::shared_ptr<AbstractCommDriver>>& drivers =
99  registry.GetDrivers();
100 
101  for (auto& driver : drivers) {
102  if (driver->bus == NavAddr::Bus::N0183) {
103  ConnectionParams params;
104  auto drv_serial =
105  std::dynamic_pointer_cast<CommDriverN0183Serial>(driver);
106  if (drv_serial) {
107  params = drv_serial->GetParams();
108  } else {
109  auto drv_net = std::dynamic_pointer_cast<CommDriverN0183Net>(driver);
110  if (drv_net) {
111  params = drv_net->GetParams();
112  }
113 #ifdef __ANDROID__
114  else {
115  auto drv_bluetooth =
116  std::dynamic_pointer_cast<CommDriverN0183AndroidBT>(driver);
117  if (drv_bluetooth) {
118  params = drv_bluetooth->GetParams();
119  }
120  }
121 #endif
122  }
123 
124  if (params.IOSelect == DS_TYPE_INPUT_OUTPUT ||
125  params.IOSelect == DS_TYPE_OUTPUT) {
126  bool bout_filter = params.SentencePassesFilter(msg, FILTER_OUTPUT);
127  if (bout_filter) {
128  std::string id = msg.ToStdString().substr(1, 5);
129  auto msg_out = std::make_shared<Nmea0183Msg>(
130  id, msg.ToStdString(),
131  std::make_shared<NavAddr0183>(driver->iface));
132 
133  bool bxmit_ok = driver->SendMessage(
134  msg_out, std::make_shared<NavAddr0183>(driver->iface));
135 
136  if (bxmit_ok)
137  LogBroadcastOutputMessageColor(msg, params.GetDSPort(), "<BLUE>",
138  nmea_log);
139  else
140  LogBroadcastOutputMessageColor(msg, params.GetDSPort(), "<RED>",
141  nmea_log);
142  } else
143  LogBroadcastOutputMessageColor(msg, params.GetDSPort(), "<CORAL>",
144  nmea_log);
145  }
146  }
147  }
148  // Send to plugins
149  on_msg_sent.Notify(msg.ToStdString());
150 }

152 std::shared_ptr<AbstractCommDriver> CreateOutputConnection(
153  const wxString& com_name, ConnectionParams& params_save, bool& btempStream,
154  bool& b_restoreStream, N0183DlgCtx dlg_ctx) {
155  std::shared_ptr<AbstractCommDriver> driver;
156  auto& registry = CommDriverRegistry::GetInstance();
157  const std::vector<std::shared_ptr<AbstractCommDriver>>& drivers =
158  registry.GetDrivers();
159 
160  int baud = 0;
161  wxString comx;
162  bool bGarmin = false;
163  if (com_name.Lower().StartsWith("serial")) {
164  comx = com_name.AfterFirst(':'); // strip "Serial:"
165  comx =
166  comx.BeforeFirst(' '); // strip off any description provided by Windows
167  std::shared_ptr<AbstractCommDriver> old_driver;
168  old_driver = FindDriver(drivers, comx.ToStdString());
169  wxLogDebug("Looking for old stream %s", com_name);
170 
171  if (old_driver) {
172  auto drv_serial_n0183 =
173  std::dynamic_pointer_cast<CommDriverN0183Serial>(old_driver);
174  if (drv_serial_n0183) {
175  params_save = drv_serial_n0183->GetParams();
176  baud = params_save.Baudrate;
177  bGarmin = params_save.Garmin;
178  }
179  registry.Deactivate(old_driver);
180 
181  b_restoreStream = true;
182  }
183 
184  if (baud == 0) baud = 4800;
185  }
186  if (com_name.Lower().StartsWith("serial")) {
187  ConnectionParams cp;
188  cp.Type = SERIAL;
189  cp.SetPortStr(comx);
190  cp.Baudrate = baud;
191  //cp.Garmin = bGarmin;
192  cp.IOSelect = DS_TYPE_OUTPUT;
193 
194  driver = MakeCommDriver(&cp);
195  btempStream = true;
196 
197 #ifdef __ANDROID__
198  wxMilliSleep(1000);
199 #else
200  auto drv_serial_n0183 =
201  std::dynamic_pointer_cast<CommDriverN0183Serial>(driver);
202  if (drv_serial_n0183) {
203  if ((wxNOT_FOUND != com_name.Upper().Find("USB")) &&
204  (wxNOT_FOUND != com_name.Upper().Find("GARMIN"))) {
205  // Wait up to 1 seconds for serial Driver secondary thread to come up
206  int timeout = 0;
207  while (!drv_serial_n0183->IsGarminThreadActive() && (timeout < 50)) {
208  wxMilliSleep(100);
209  wxYield();
210  timeout++;
211  }
212 
213  if (!drv_serial_n0183->IsGarminThreadActive()) {
214  MESSAGE_LOG << "-->GPS Port:" << com_name
215  << " ...Could not be opened for writing";
216  }
217  } else {
218  // Wait up to 1 seconds for serial Driver secondary thread to come up
219  int timeout = 0;
220  while (!drv_serial_n0183->IsSecThreadActive() && (timeout < 50)) {
221  wxMilliSleep(100);
222  timeout++;
223  }
224 
225  if (!drv_serial_n0183->IsSecThreadActive()) {
226  MESSAGE_LOG << "-->GPS Port:" << com_name
227  << " ...Could not be opened for writing";
228  }
229  }
230  }
231 #endif
232  } else
233  driver = FindDriver(drivers, com_name.ToStdString());
234 
235  if (com_name.Find("Bluetooth") != wxNOT_FOUND) {
236  wxString comm_addr = com_name.AfterFirst(';');
237 
238  driver = FindDriver(drivers, comm_addr.ToStdString());
239  if (!driver) {
240  // Force Android Bluetooth to use only already enabled driver
241  return driver;
242 
243  ConnectionParams ConnectionParams;
244  ConnectionParams.Type = INTERNAL_BT;
245  wxStringTokenizer tkz(com_name, ";");
246  wxString name = tkz.GetNextToken();
247  wxString mac = tkz.GetNextToken();
248 
249  ConnectionParams.NetworkAddress = name;
250  ConnectionParams.Port = mac;
251  ConnectionParams.NetworkPort = 0;
252  ConnectionParams.NetProtocol = PROTO_UNDEFINED;
253  ConnectionParams.Baudrate = 0;
254 
255  driver = MakeCommDriver(&ConnectionParams);
256 
257  btempStream = true;
258  }
259  } else if (com_name.Lower().StartsWith("udp") ||
260  com_name.Lower().StartsWith("tcp")) {
261  std::shared_ptr<CommDriverN0183Net> drv_net_n0183;
262 
263  if (!driver) {
264  NetworkProtocol protocol = UDP;
265  if (com_name.Lower().StartsWith("tcp")) protocol = TCP;
266  wxStringTokenizer tkz(com_name, ":");
267  wxString token = tkz.GetNextToken();
268  wxString address = tkz.GetNextToken();
269  token = tkz.GetNextToken();
270  long port;
271  token.ToLong(&port);
272 
273  ConnectionParams cp;
274  cp.Type = NETWORK;
275  cp.NetProtocol = protocol;
276  cp.NetworkAddress = address;
277  cp.NetworkPort = port;
278  cp.IOSelect = DS_TYPE_INPUT_OUTPUT;
279 
280  driver = MakeCommDriver(&cp);
281  btempStream = true;
282  }
283  drv_net_n0183 = std::dynamic_pointer_cast<CommDriverN0183Net>(driver);
284 
285  if (com_name.Lower().StartsWith("tcp")) {
286  // new tcp connections must wait for connect
287  std::string msg(_("Connecting to "));
288  msg += com_name;
289  dlg_ctx.set_message(msg);
290  dlg_ctx.pulse();
291 
292  if (drv_net_n0183) {
293  int loopCount = 10; // seconds
294  bool bconnected = false;
295  while (!bconnected && (loopCount > 0)) {
296  if (drv_net_n0183->GetSock()->IsConnected()) {
297  bconnected = true;
298  break;
299  }
300  dlg_ctx.pulse();
301  wxYield();
302  wxSleep(1);
303  loopCount--;
304  }
305 
306  if (bconnected) {
307  msg = _("Connected to ");
308  msg += com_name;
309  dlg_ctx.set_message(msg);
310  } else {
311  if (btempStream) {
312  registry.Deactivate(driver);
313  }
314  return 0;
315  }
316  }
317  }
318  }
319  return driver;
320 }

587  // Create the NMEA Rte sentence
588  // Try to create a single sentence, and then check the length to see if
589  // too long
590  unsigned int max_length = 76;
591  unsigned int max_wp = 2; // seems to be required for garmin...
592 
593  // Furuno GPS can only accept 5 (five) waypoint linkage sentences....
594  // So, we need to compact a few more points into each link sentence.
595  if (g_GPS_Ident == "FurunoGP3X") {
596  max_wp = 8;
597  max_length = 80;
598  }
599 
600  // Furuno has its own talker ID, so do not allow the global override
601  wxString talker_save = g_TalkerIdText;
602  if (g_GPS_Ident == "FurunoGP3X") g_TalkerIdText.Clear();
603 
604  oNMEA0183.Rte.Empty();
605  oNMEA0183.Rte.TypeOfRoute = CompleteRoute;
606 
607  if (pr->m_RouteNameString.IsEmpty())
608  oNMEA0183.Rte.RouteName = _T ( "1" );
609  else
610  oNMEA0183.Rte.RouteName = pr->m_RouteNameString;
611 
612  if (g_GPS_Ident == "FurunoGP3X") {
613  oNMEA0183.Rte.RouteName = _T ( "01" );
614  oNMEA0183.TalkerID = _T ( "GP" );
615  oNMEA0183.Rte.m_complete_char = 'C'; // override the default "c"

617  }

620  oNMEA0183.Rte.message_number = 1;
621 
622  // add the waypoints
623  auto node = pr->pRoutePointList->GetFirst();
624  while (node) {
625  RoutePoint* prp = node->GetData();
626  wxString name = prp->GetName().Truncate(g_maxWPNameLength);
627 
628  if (g_GPS_Ident == "FurunoGP3X") {
629  name = prp->GetName();
630  name += "000000";
631  name.Truncate(g_maxWPNameLength);
632  name.Prepend(" "); // What Furuno calls "Skip Code", space means
633  // use the WP
634  }
635 
636  oNMEA0183.Rte.AddWaypoint(name);
637  node = node->GetNext();
638  }
639 
640  oNMEA0183.Rte.Write(snt);
641 
642  if ((snt.Sentence.Len() > max_length) ||
643  (pr->pRoutePointList->GetCount() >
