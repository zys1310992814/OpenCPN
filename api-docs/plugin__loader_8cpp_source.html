118 static std::string GetInstalledVersion(const PlugInData& pd) {
119  std::string path =
120  PluginHandler::versionPath(pd.m_common_name.ToStdString());
121  if (path == "" || !wxFileName::IsFileReadable(path)) {
122  auto loader = PluginLoader::getInstance();
123  auto pic = GetContainer(pd, *loader->GetPlugInArray());
124  if (!pic || !pic->m_pplugin) {
125  return SemanticVersion(0, 0, -1).to_string();
126  }
127  int v_major = pic->m_pplugin->GetPlugInVersionMajor();
128  int v_minor = pic->m_pplugin->GetPlugInVersionMinor();
129  return SemanticVersion(v_major, v_minor, -1).to_string();
130  }
131  std::ifstream stream;
132  std::string version;
133  stream.open(path, std::ifstream::in);
134  stream >> version;
135  return version;
136 }

154 std::string PluginLoader::GetPluginVersion(
155  const PlugInData pd,
156  std::function<const PluginMetadata(const std::string&)> get_metadata) {
157  auto loader = PluginLoader::getInstance();
158  auto pic = GetContainer(pd, *loader->GetPlugInArray());
159  if (!pic) {
160  return SemanticVersion(0, 0, -1).to_string();
161  }
162 
163  PluginMetadata metadata;
164  metadata = pic->m_managed_metadata;
165  if (metadata.version == "")
166  metadata = get_metadata(pic->m_common_name.ToStdString());
167  std::string detail_suffix(metadata.is_imported ? _(" [Imported]") : "");
168  if (metadata.is_orphan)
169  detail_suffix = _(" [Orphan]");
170 
171  int v_major(0);
172  int v_minor(0);
173  if (pic->m_pplugin) {
174  v_major = pic->m_pplugin->GetPlugInVersionMajor();
175  v_minor = pic->m_pplugin->GetPlugInVersionMinor();
176  }
177  auto p = dynamic_cast<opencpn_plugin_117*>(pic->m_pplugin);
178  if (p) {
179  // New style plugin, trust version available in the API.
180  auto sv = SemanticVersion(
181  v_major, v_minor, p->GetPlugInVersionPatch(), p->GetPlugInVersionPost(),
182  p->GetPlugInVersionPre(), p->GetPlugInVersionBuild());
183  return sv.to_string() + detail_suffix;
184  } else {
185  if (!metadata.is_orphan) {
186  std::string version = GetInstalledVersion(pd);
187  return version + detail_suffix;
188  }
189  else
190  return metadata.version + detail_suffix;
191  }
192 }

238 static void setLoadPath() {
239  using namespace std;
240 
241  auto const osSystemId = wxPlatformInfo::Get().GetOperatingSystemId();
242  auto dirs = PluginPaths::getInstance()->Libdirs();
243  if (osSystemId & wxOS_UNIX_LINUX) {
244  string path = ocpn::join(dirs, ':');
245  wxString envPath;
246  if (wxGetEnv("LD_LIBRARY_PATH", &envPath)) {
247  path = path + ":" + envPath.ToStdString();
248  }
249  wxLogMessage("Using LD_LIBRARY_PATH: %s", path.c_str());
250  wxSetEnv("LD_LIBRARY_PATH", path.c_str());
251  } else if (osSystemId & wxOS_WINDOWS) {
252  // On windows, Libdirs() and Bindirs() are the same.
253  string path = ocpn::join(dirs, ';');
254  wxString envPath;
255  if (wxGetEnv("PATH", &envPath)) {
256  path = path + ";" + envPath.ToStdString();
257  }
258  wxLogMessage("Using PATH: %s", path);
259  wxSetEnv("PATH", path);
260  } else if (osSystemId & wxOS_MAC) {
261  string path = ocpn::join(dirs, ':');
262  wxString envPath;
263  if (wxGetEnv("DYLD_LIBRARY_PATH", &envPath)) {
264  path = path + ":" + envPath.ToStdString();
265  }
266  wxLogMessage("Using DYLD_LIBRARY_PATH: %s", path.c_str());
267  wxSetEnv("DYLD_LIBRARY_PATH", path.c_str());
268  } else {
269  wxString os_name = wxPlatformInfo::Get().GetPortIdName();
270  if (os_name.Contains("wxQT")) {
271  wxLogMessage("setLoadPath() using Android library path");
272  } else
273  wxLogWarning("SetLoadPath: Unsupported platform.");
274  }
275  if (osSystemId & wxOS_MAC || osSystemId & wxOS_UNIX_LINUX) {
276  dirs = PluginPaths::getInstance()->Bindirs();
277  string path = ocpn::join(dirs, ':');
278  wxString envPath;
279  wxGetEnv("PATH", &envPath);
280  path = path + ":" + envPath.ToStdString();
281  wxLogMessage("Using PATH: %s", path);
282  wxSetEnv("PATH", path);
283  }
284 }

429  // This only needs to happen when the entire universe (enabled and disabled)

434  evt_update_chart_types.Notify();
435  auto errors = std::make_shared<std::vector<LoadError>>(load_errors);
436  evt_plugin_loadall_finalize.Notify(errors, "");
437  load_errors.clear();
438 
439  return any_dir_loaded;

442 bool PluginLoader::LoadPluginCandidate(const wxString& file_name,
443  bool load_enabled) {
444  wxString plugin_file = wxFileName(file_name).GetFullName();
445  wxLogMessage("Checking plugin candidate: %s", file_name.mb_str().data());
446  wxDateTime plugin_modification = wxFileName(file_name).GetModificationTime();
447  wxLog::FlushActive();
448 
449  // this gets called every time we switch to the plugins tab.
450  // this allows plugins to be installed and enabled without restarting
451  // opencpn. For this reason we must check that we didn't already load this
452  // plugin
453  bool loaded = false;
454  PlugInContainer* loaded_pic = nullptr;
455  for (unsigned int i = 0; i < plugin_array.GetCount(); i++) {
456  PlugInContainer* pic_test = plugin_array[i];
457 
458  // Checking for dynamically updated plugins
459  if (pic_test->m_plugin_filename == plugin_file) {
460  // Do not re-load same-name plugins from different directories. Certain
461  // to crash...
462  if (pic_test->m_plugin_file == file_name) {
463  if (pic_test->m_plugin_modification != plugin_modification) {
464  // modification times don't match, reload plugin
465  plugin_array.Remove(pic_test);
466  i--;
467 
468  DeactivatePlugIn(pic_test);
469  pic_test->m_destroy_fn(pic_test->m_pplugin);
470 
471  delete pic_test;
472  } else {
473  loaded = true;
474  loaded_pic = pic_test;
475  break;
476  }
477  } else {
478  loaded = true;
479  loaded_pic = pic_test;
480  break;
481  }
482  }
483  }
484 
485  if (loaded) return true;
486 
487  // Avoid loading/testing legacy plugins installed in base plugin path.
488  wxFileName fn_plugin_file(file_name);
489  wxString plugin_file_path =
490  fn_plugin_file.GetPath(wxPATH_GET_VOLUME | wxPATH_GET_SEPARATOR);
491  wxString base_plugin_path = g_BasePlatform->GetPluginDir();
492  if (!base_plugin_path.EndsWith(wxFileName::GetPathSeparator()))
493  base_plugin_path += wxFileName::GetPathSeparator();
494 
495  if (!g_bportable) {
496  if (base_plugin_path.IsSameAs(plugin_file_path)) {
497  if (!IsSystemPluginPath(file_name.ToStdString())) {
498  DEBUG_LOG << "Skipping plugin " << file_name << " in "
499  << g_BasePlatform->GetPluginDir();
500  return false;
501  }
502  }
503  }
504 
505  if (!IsSystemPluginPath(file_name.ToStdString()) && safe_mode::get_mode()) {
506  DEBUG_LOG << "Skipping plugin " << file_name << " in safe mode";
507  return false;
508  }
509 
510  auto msg =
511  std::string("Checking plugin compatibility: ") + file_name.ToStdString();
512  wxLogMessage(msg.c_str());
513  wxLog::FlushActive();
514 
515  bool b_compat = CheckPluginCompatibility(file_name);
516 
517  if (!b_compat) {
518  msg =
519  std::string("Incompatible plugin detected: ") + file_name.ToStdString();
520  wxLogMessage(msg.c_str());
521  if (m_blacklist->mark_unloadable(file_name.ToStdString())) {
522  LoadError le(LoadError::Type::Unloadable, file_name.ToStdString());
523  load_errors.push_back(le);
524  }
525  return false;
526  }
527 
528  PlugInContainer* pic = LoadPlugIn(file_name);
529 
530  // Check the config file to see if this PlugIn is user-enabled,
531  // only loading enabled plugins.
532  // Make the check late enough to pick up incompatible plugins anyway
533  const auto path = std::string("/PlugIns/") + plugin_file.ToStdString();
534  ConfigVar<bool> enabled(path, "bEnabled", TheBaseConfig());
535  if (load_enabled && !enabled.Get(true)) {
536  pic->m_destroy_fn(pic->m_pplugin);
537  delete pic;
538  wxLogMessage("Skipping not enabled candidate.");
539  return true;
540  }
541 
542  if (pic) {
543  if (pic->m_pplugin) {
544  plugin_array.Add(pic);
545 
546  // The common name is available without initialization and startup of
547  // the PlugIn
548  pic->m_common_name = pic->m_pplugin->GetCommonName();
549  pic->m_plugin_filename = plugin_file;
550  pic->m_plugin_modification = plugin_modification;
551  pic->m_enabled = enabled.Get(false);
552 
553  if (safe_mode::get_mode()) {
554  pic->m_enabled = false;
555  enabled.Set(false);
556  }
557  if (dynamic_cast<wxApp*>(wxAppConsole::GetInstance())) {
558  // The CLI has no graphics context, but plugins assumes there is.
559  if (pic->m_enabled) {
560  pic->m_cap_flag = pic->m_pplugin->Init();
561  pic->m_init_state = true;
562  }
563  }
564  evt_load_plugin.Notify(pic);
565  wxLog::FlushActive();
566 
567  std::string found_version;
568  for (const auto& p : PluginHandler::getInstance()->getInstalled()) {
569  if (p.name == pic->m_common_name.Lower()) {
570  found_version = p.readonly ? "" : p.version;
571  break;
572  }
573  }
574  pic->m_version_str = found_version;
575  pic->m_short_description = pic->m_pplugin->GetShortDescription();
576  pic->m_long_description = pic->m_pplugin->GetLongDescription();
577  pic->m_version_major = pic->m_pplugin->GetPlugInVersionMajor();
578  pic->m_version_minor = pic->m_pplugin->GetPlugInVersionMinor();
579 
580  auto pbm0 = pic->m_pplugin->GetPlugInBitmap();
581  if (!pbm0->IsOk()) {
582  pbm0 = (wxBitmap *)GetPluginDefaultIcon();
583  }
584  pic->m_bitmap = wxBitmap(pbm0->GetSubBitmap(
585  wxRect(0, 0, pbm0->GetWidth(), pbm0->GetHeight())));
586 
587  if (!pic->m_enabled && pic->m_destroy_fn) {
588  pic->m_destroy_fn(pic->m_pplugin);
589  pic->m_destroy_fn = nullptr;
590  pic->m_pplugin = nullptr;
591  pic->m_init_state = false;
592  if (pic->m_library.IsLoaded()) pic->m_library.Unload();
593  }
594 
595  // Check to see if the plugin just processed has an associated catalog entry
596  // understanding that SYSTEM plugins have no metadata by design
597  auto found = std::find(SYSTEM_PLUGINS.begin(), SYSTEM_PLUGINS.end(),
598  pic->m_common_name.Lower());
599  bool is_system = found != SYSTEM_PLUGINS.end();
600 
601  if (!is_system) {
602  auto available = PluginHandler::getInstance()->getCompatiblePlugins();
603  wxString name = pic->m_common_name;
604  auto it = find_if(available.begin(), available.end(),
605  [name](const PluginMetadata& md) { return md.name == name; });
606 
607  if (it == available.end()) {
608  // Installed plugin is an orphan....
609  // Add a stub metadata entry to the active CatalogHandler context
610  // to satisfy minimal PIM functionality
611 
612  auto oprhan_metadata = CreateMetadata(pic);
613  auto catalogHdlr = CatalogHandler::getInstance();
614  catalogHdlr->AddMetadataToActiveContext(oprhan_metadata);
615  }
616  }
617 
618  } else { // No pic->m_pplugin
619  wxLogMessage(
620  " PluginLoader: Unloading invalid PlugIn, API version %d ",
621  pic->m_api_version);
622  pic->m_destroy_fn(pic->m_pplugin);
623 
624  LoadError le(LoadError::Type::Unloadable, file_name.ToStdString());
625  delete pic;
626  load_errors.push_back(le);
627  return false;
628  }
629  } else { // pic == 0
630  return false;
631  }
632  return true;
633 }

636 bool PluginLoader::LoadPlugInDirectory(const wxString& plugin_dir,
637  bool load_enabled) {
638  evt_load_directory.Notify();
639  m_plugin_location = plugin_dir;
640 
641  wxString msg("PluginLoader searching for PlugIns in location ");
642  msg += m_plugin_location;
643  wxLogMessage(msg);
644 
645 #ifdef __WXMSW__
646  wxString pispec = "*_pi.dll";
647 #elif defined(__WXOSX__)
648  wxString pispec = "*_pi.dylib";
649 #else
650  wxString pispec = "*_pi.so";
651 #endif
652 
653  if (!::wxDirExists(m_plugin_location)) {
654  msg = m_plugin_location;
655  msg.Prepend(" Directory ");
656  msg.Append(" does not exist.");
657  wxLogMessage(msg);
658  return false;
659  }
660

884  std::vector<PlugInContainer*> loaded_plugins;
885  for (size_t i = 0; i < plugin_array.GetCount(); i++)
886  loaded_plugins.push_back(plugin_array.Item(i));

891  p->m_common_name.Lower().ToStdString());
892  bool is_system = found != SYSTEM_PLUGINS.end();
893  p->m_status = is_system ? PluginStatus::System : PluginStatus::Unmanaged;
894  }

897  // available in the current catalog Usually due to reverting from Alpha/Beta
898  // catalog back to master
899  auto predicate = [](const PlugInContainer* pd) -> bool {
900  const auto md(
901  PluginLoader::MetadataByName(pd->m_common_name.ToStdString()));

1012  GElf_Ehdr elf_file_header;
1013  Elf_Scn* elf_section_handle = nullptr;
1014 
1015  file_handle = open(file, O_RDONLY);
1016  if (file_handle == -1) {
1017  wxLogMessage("Could not open file \"%s\" for reading: %s", file,
1018  strerror(errno));
1019  goto FailureEpilogue;
1020  }
1021

1023  if (elf_handle == nullptr) {

1034  case ET_EXEC:
1035  case ET_DYN:
1036  break;

1039  "Module \"%s\" is not an executable or shared library.", file));
1040  goto FailureEpilogue;
1041  }

1044  (static_cast<uint64_t>(elf_file_header.e_ident[EI_CLASS])
1045  << 0) | // ELF class (32/64).
1046  (static_cast<uint64_t>(elf_file_header.e_ident[EI_DATA])
1047  << 8) | // Endianness.
1048  (static_cast<uint64_t>(elf_file_header.e_ident[EI_OSABI])
1049  << 16) | // OS ABI (Linux, FreeBSD, etc.).
1050  (static_cast<uint64_t>(elf_file_header.e_ident[EI_ABIVERSION])
1051  << 24) | // OS ABI version.
1052  (static_cast<uint64_t>(elf_file_header.e_machine)
1053  << 32) | // Instruction set.
1054  0;
1055 
1056  while ((elf_section_handle = elf_nextscn(elf_handle, elf_section_handle)) !=
1057  nullptr) {
1058  GElf_Shdr elf_section_header;
1059  Elf_Data* elf_section_data = nullptr;
1060  size_t elf_section_entry_count = 0;
1061

1063  &elf_section_header) {
1064  wxLogMessage("Could not get ELF section header from \"%s\".", file);
1065  goto FailureEpilogue;
1066  } else if (elf_section_header.sh_type != SHT_DYNAMIC) {
1067  continue;
1068  }
1069 
1070  elf_section_data = elf_getdata(elf_section_handle, nullptr);
1071  if (elf_section_data == nullptr) {
1072  wxLogMessage("Could not get ELF section data from \"%s\".", file);
1073  goto FailureEpilogue;
1074  }
1075 
1076  if ((elf_section_data->d_size == 0) ||
1077  (elf_section_header.sh_entsize == 0)) {
1078  wxLogMessage("Got malformed ELF section metadata from \"%s\".", file);
1079  goto FailureEpilogue;
1080  }
1081 
1082  elf_section_entry_count =
1083  elf_section_data->d_size / elf_section_header.sh_entsize;
1084  for (size_t elf_section_entry_index = 0;
1085  elf_section_entry_index < elf_section_entry_count;
1086  ++elf_section_entry_index) {
1087  GElf_Dyn elf_dynamic_entry;
1088  const char* elf_dynamic_entry_name = nullptr;
1089  if (gelf_getdyn(elf_section_data,
1090  static_cast<int>(elf_section_entry_index),
1091  &elf_dynamic_entry) != &elf_dynamic_entry) {
1092  wxLogMessage("Could not get ELF dynamic_section entry from \"%s\".",
1093  file);
1094  goto FailureEpilogue;

1096  continue;
1097  }
1098  elf_dynamic_entry_name = elf_strptr(
1099  elf_handle, elf_section_header.sh_link, elf_dynamic_entry.d_un.d_val);
1100  if (elf_dynamic_entry_name == nullptr) {
1101  wxLogMessage(wxString::Format("Could not get %s %s from \"%s\".", "ELF",
1102  "string entry", file));
1103  goto FailureEpilogue;
1104  }

1106  wxString name_part(elf_dynamic_entry_name,
