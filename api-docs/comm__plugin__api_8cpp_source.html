55 std::shared_ptr<void> GetSignalkPayload(ObservedEvt ev) {
56  auto msg = UnpackEvtPointer<SignalkMsg>(ev);
57  wxJSONReader reader;
58  wxJSONValue data;
59  reader.Parse(wxString(msg->raw_message), &data);
60 
61  wxJSONValue root(wxJSONTYPE_OBJECT);
62  root["Data"] = data;
63  root["ErrorCount"] = reader.GetErrorCount();
64  root["WarningCount"] = reader.GetWarningCount();
65 
66  root["Errors"] = wxJSONValue(wxJSONTYPE_ARRAY);
67  for (size_t i = 0; i < reader.GetErrors().GetCount(); i++)
68  root["Errors"].Append(reader.GetErrors().Item(i));
69 
70  root["Warnings"] = wxJSONValue(wxJSONTYPE_ARRAY);
71  for (size_t i = 0; i < reader.GetWarnings().GetCount(); i++)
72  root["Warnings"].Append(reader.GetWarnings().Item(i));
73 
74  root["Context"] = msg->context;
75  root["ContextSelf"] = msg->context_self;
76 
77  return static_pointer_cast<void>(std::make_shared<wxJSONValue>(root));
78 }
