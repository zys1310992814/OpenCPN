230 static wxPoint transrot(wxPoint pt, float sin_theta, float cos_theta,
231  wxPoint offset = wxPoint(0, 0)) {
232  wxPoint ret;
233  float px = (float)(pt.x * sin_theta) + (float)(pt.y * cos_theta);
234  float py = (float)(pt.y * sin_theta) - (float)(pt.x * cos_theta);
235  ret.x = (int)wxRound(px);
236  ret.y = (int)wxRound(py);
237  ret.x += offset.x;
238  ret.y += offset.y;
239 
240  return ret;
241 }

249 void AISDrawAreaNotices(ocpnDC &dc, ViewPort &vp, ChartCanvas *cp) {
250  if (cp == NULL) return;
251  if (!g_pAIS || !cp->GetShowAIS() || !g_bShowAreaNotices) return;
252 
253  wxDateTime now = wxDateTime::Now();
254  now.MakeGMT();
255 
256  bool b_pens_set = false;
257  wxPen pen_save;
258  wxBrush brush_save;
259  wxColour yellow;
260  wxColour green;
261  wxPen pen;
262  wxBrush *yellow_brush = wxTheBrushList->FindOrCreateBrush(
263  wxColour(0, 0, 0), wxBRUSHSTYLE_TRANSPARENT);
264  wxBrush *green_brush = wxTheBrushList->FindOrCreateBrush(
265  wxColour(0, 0, 0), wxBRUSHSTYLE_TRANSPARENT);
266  ;
267  wxBrush *brush;
268 
269  float vp_scale = vp.view_scale_ppm;
270 
271  for (const auto &target : g_pAIS->GetAreaNoticeSourcesList()) {
272  auto target_data = target.second;
273  if (!target_data->area_notices.empty()) {
274  if (!b_pens_set) {
275  pen_save = dc.GetPen();
276  brush_save = dc.GetBrush();
277 
278  yellow = GetGlobalColor(_T ( "YELO1" ));
279  yellow.Set(yellow.Red(), yellow.Green(), yellow.Blue(), 64);
280 
281  green = GetGlobalColor(_T ( "GREEN4" ));
282  green.Set(green.Red(), green.Green(), green.Blue(), 64);
283 
284  pen.SetColour(yellow);
285  pen.SetWidth(2);
286 
287  yellow_brush = wxTheBrushList->FindOrCreateBrush(
288  yellow, wxBRUSHSTYLE_CROSSDIAG_HATCH);
289  green_brush =
290  wxTheBrushList->FindOrCreateBrush(green, wxBRUSHSTYLE_TRANSPARENT);
291  brush = yellow_brush;
292 
293  b_pens_set = true;
294  }
295 
296  for (auto &ani : target_data->area_notices) {
297  Ais8_001_22 &area_notice = ani.second;
298 
299  if (area_notice.expiry_time > now) {
300  std::vector<wxPoint> points;
301  bool draw_polygon = false;
302 
303  switch (area_notice.notice_type) {
304  case 0:
305  pen.SetColour(green);
306  brush = green_brush;
307  break;
308  case 1:
309  pen.SetColour(yellow);
310  brush = yellow_brush;
311  break;
312  default:
313  pen.SetColour(yellow);
314  brush = yellow_brush;
315  }
316  dc.SetPen(pen);
317  dc.SetBrush(*brush);
318 
319  for (Ais8_001_22_SubAreaList::iterator sa =
320  area_notice.sub_areas.begin();
321  sa != area_notice.sub_areas.end(); ++sa) {
322  switch (sa->shape) {
323  case AIS8_001_22_SHAPE_CIRCLE: {
324  wxPoint target_point;
325  GetCanvasPointPix(vp, cp, sa->latitude, sa->longitude,
326  &target_point);
327  points.push_back(target_point);
328  if (sa->radius_m > 0.0)
329  dc.DrawCircle(target_point, sa->radius_m * vp_scale);
330  break;
331  }
332  case AIS8_001_22_SHAPE_POLYGON:
333  draw_polygon = true;
334  // FALL THROUGH
335  case AIS8_001_22_SHAPE_POLYLINE: {
336  double lat = sa->latitude;
337  double lon = sa->longitude;
338  for (int i = 0; i < 4; ++i) {
339  ll_gc_ll(lat, lon, sa->angles[i], sa->dists_m[i] / 1852.0,
340  &lat, &lon);
341  wxPoint target_point;
342  GetCanvasPointPix(vp, cp, lat, lon, &target_point);
343  points.push_back(target_point);
344  }
345  }
346  }
347  }
348  if (draw_polygon) dc.DrawPolygon(points.size(), &points.front());
349  }
350  }
351  }
352  }
353 
354  if (b_pens_set) {
355  dc.SetPen(pen_save);
356  dc.SetBrush(brush_save);
357  }
358 }

617 // spherical coordinates is sufficient for visually plotting with relatively
618 // small distances and about 6x faster than ll_gc_ll
619 static void spherical_ll_gc_ll(float lat, float lon, float brg, float dist,
620  float *dlat, float *dlon) {
621  float angr = brg / 180 * M_PI;
622  float latr = lat * M_PI / 180;
623  float D = dist / 3443; // earth radius in nm
624  float sD = sinf(D), cD = cosf(D);
625  float sy = sinf(latr), cy = cosf(latr);
626  float sa = sinf(angr), ca = cosf(angr);
627 
628  *dlon = lon + asinf(sa * sD / cy) * 180 / M_PI;
629  *dlat = asinf(sy * cD + cy * sD * ca) * 180 / M_PI;
630 }
