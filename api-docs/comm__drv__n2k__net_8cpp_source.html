107 void circular_buffer::put(unsigned char item) {
108  std::lock_guard<std::mutex> lock(mutex_);
109  buf_[head_] = item;
110  if (full_) tail_ = (tail_ + 1) % max_size_;
111 
112  head_ = (head_ + 1) % max_size_;
113 
114  full_ = head_ == tail_;
115 }

117 unsigned char circular_buffer::get() {
118  std::lock_guard<std::mutex> lock(mutex_);
119 
120  if (empty()) return 0;
121 
122  // Read data and advance the tail (we now have a free space)
123  auto val = buf_[tail_];
124  full_ = false;
125  tail_ = (tail_ + 1) % max_size_;
126 
127  return val;
128 }

292 void CommDriverN2KNet::OpenNetworkUDP(unsigned int addr) {
293  if (GetPortType() != DS_TYPE_OUTPUT) {
294  // We need a local (bindable) address to create the Datagram receive socket
295  // Set up the receive socket
296  wxIPV4address conn_addr;
297  conn_addr.Service(GetNetPort());
298  conn_addr.AnyAddress();
299  SetSock(
300  new wxDatagramSocket(conn_addr, wxSOCKET_NOWAIT | wxSOCKET_REUSEADDR));
301 
302  // Test if address is IPv4 multicast
303  if ((ntohl(addr) & 0xf0000000) == 0xe0000000) {
304  SetMulticast(true);
305  m_mrq_container->SetMrqAddr(addr);
306  GetSock()->SetOption(IPPROTO_IP, IP_ADD_MEMBERSHIP, &m_mrq_container->m_mrq,
307  sizeof(m_mrq_container->m_mrq));
308  }
309 
310  GetSock()->SetEventHandler(*this, DS_SOCKET_ID);
311 
312  GetSock()->SetNotify(wxSOCKET_CONNECTION_FLAG | wxSOCKET_INPUT_FLAG |
313  wxSOCKET_LOST_FLAG);
314  GetSock()->Notify(TRUE);
315  GetSock()->SetTimeout(1); // Short timeout
316  }
317 
318  // Set up another socket for transmit
319  if (GetPortType() != DS_TYPE_INPUT) {
320  wxIPV4address tconn_addr;
321  tconn_addr.Service(0); // use ephemeral out port
322  tconn_addr.AnyAddress();
323  SetTSock(
324  new wxDatagramSocket(tconn_addr, wxSOCKET_NOWAIT | wxSOCKET_REUSEADDR));
325  // Here would be the place to disable multicast loopback
326  // but for consistency with broadcast behaviour, we will
327  // instead rely on setting priority levels to ignore
328  // sentences read back that have just been transmitted
329  if ((!GetMulticast()) && (GetAddr().IPAddress().EndsWith(_T("255")))) {
330  int broadcastEnable = 1;
331  bool bam = GetTSock()->SetOption(
332  SOL_SOCKET, SO_BROADCAST, &broadcastEnable, sizeof(broadcastEnable));
333  }
334  }
335 
336  // In case the connection is lost before acquired....
337  SetConnectTime(wxDateTime::Now());
338 }

342  wxLogMessage(wxString::Format(_T("Opening TCP Server %d"), isServer));
343

352  GetSockServer()->SetNotify(wxSOCKET_CONNECTION_FLAG);

360  GetSock()->SetNotify(notify_flags);
361  GetSock()->Notify(TRUE);
362  GetSock()->SetTimeout(1); // Short timeout

364  SetBrxConnectEvent(false);
365  GetSocketTimer()->Start(100, wxTIMER_ONE_SHOT); // schedule a connection
366  }

468 void CommDriverN2KNet::HandleCanFrameInput(can_frame frame) {
469  int position = -1;
470  bool ready = true;
471 
472  CanHeader header(frame);
473  if (header.IsFastMessage()) {
474  position = fast_messages->FindMatchingEntry(header, frame.data[0]);
475  if (position == kNotFound) {
476  // Not an existing fast message: create new entry and insert first frame
477  position = fast_messages->AddNewEntry();
478  ready = fast_messages->InsertEntry(header, frame.data, position);
479  } else {
480  // An existing fast message entry is present, append the frame
481  ready = fast_messages->AppendEntry(header, frame.data, position);
482  }
483  }
484  if (ready) {
485  std::vector<unsigned char> vec;
486  if (position >= 0) {
487  // Re-assembled fast message
488  vec = PushFastMsgFragment(header, position);
489  } else {
490  // Single frame message
491  vec = PushCompleteMsg(header, position, frame);
492  }
493 
494  // Message is ready
495  CommDriverN2KNetEvent Nevent(wxEVT_COMMDRIVER_N2K_NET, 0);
496  auto payload = std::make_shared<std::vector<uint8_t> >(vec);
497  Nevent.SetPayload(payload);
498  AddPendingEvent(Nevent);
499 
500  }
501 }

510 N2K_Format CommDriverN2KNet::DetectFormat(std::vector<unsigned char> packet) {
511 
512  // A simplistic attempt at identifying which of the various available
513  // on-wire (or air) formats being emitted by a configured
514  // Actisense N2k<->ethernet device.
515 
516  if (isASCII(packet)) {
517  if (std::find(packet.begin(), packet.end(), ':') != packet.end())
518  return N2KFormat_Actisense_RAW_ASCII;
519  else
520  return N2KFormat_Actisense_N2K_ASCII;
521  }
522  else {
523  if (packet[2] == 0x95)
524  return N2KFormat_Actisense_RAW;
525  else if (packet[2] == 0xd0)
526  return N2KFormat_Actisense_N2K;
527  }
528  return N2KFormat_Undefined;
529 }

531 bool CommDriverN2KNet::ProcessActisense_N2K(std::vector<unsigned char> packet) {
532 
533  //1002 d0 1500ff0401f80900684c1b00a074eb14f89052d288 1003
534 
535  std::vector<unsigned char> data;
536  bool bInMsg = false;
537  bool bGotESC = false;
538  bool bGotSOT = false;
539 
540  while (!m_circle->empty()) {
541  uint8_t next_byte = m_circle->get();
542 
543  if (bInMsg) {
544  if (bGotESC) {
545  if (ESCAPE == next_byte) {
546  data.push_back(next_byte);
547  bGotESC = false;
548  }
549  }
550 
551  if (bGotESC && (ENDOFTEXT == next_byte)) {
552  // Process packet
553  // first 3 bytes are: 1 byte for message type, 2 bytes for rest of message length
554  unsigned int msg_length = (uint32_t)data[1] + ((uint32_t)data[2]<<8);
555 
556  // As a sanity check, verify message length
557  if (msg_length == data.size()-1) {
558  uint8_t destination = data[3];
559  uint8_t source = data[4];
560 
561  uint8_t dprp = data[7];
562  uint8_t priority = (dprp >> 2) & 7; // priority bits are 3,4,5th bit
563  uint8_t rAndDP = dprp & 3; // data page + reserved is first 2 bits
564 
565  // PGN
566  uint8_t pduFormat = data[6]; // PF (PDU Format)
567  uint32_t pgn = (rAndDP << 16) + (pduFormat << 8);
568  if (pduFormat >=
569  240) // message is broadcast, PS contains group extension
570  pgn += data[5]; // +PS (PDU Specific)
571 
572  // Create the OCPN payload
573  std::vector<uint8_t> o_payload;
574  o_payload.push_back(0x93);
575  o_payload.push_back(0x13);
576  o_payload.push_back(priority); // priority;
577  o_payload.push_back(pgn & 0xFF);
578  o_payload.push_back((pgn >> 8) & 0xFF);
579  o_payload.push_back((pgn >> 16) & 0xFF);
580  o_payload.push_back(destination); // destination;
581  o_payload.push_back(source); // source);
582  o_payload.push_back(0xFF); // FIXME (dave) generate the time fields
583  o_payload.push_back(0xFF);
584  o_payload.push_back(0xFF);
585  o_payload.push_back(0xFF);
586  o_payload.push_back(data.size());
587 
588  // Data starts at offset 13
589  for (size_t n = 13; n < data.size() - 1; n++)
590  o_payload.push_back(data[n]);
591 
592  o_payload.push_back(0x55); // CRC dummy, not checked
593 
594  // Message is ready
595  CommDriverN2KNetEvent Nevent(wxEVT_COMMDRIVER_N2K_NET, 0);
596  auto n2k_payload = std::make_shared<std::vector<uint8_t>>(o_payload);
597  Nevent.SetPayload(n2k_payload);
598  AddPendingEvent(Nevent);
599  }
600 
601  // reset for next packet
602  bInMsg = false;
603  bGotESC = false;
604  data.clear();
605 
606  } else {
607  bGotESC = (next_byte == ESCAPE);
608 
609  if (!bGotESC) {
610  data.push_back(next_byte);
611  }
612  }
613  }
614 
615  else {
616  if (STARTOFTEXT == next_byte) {
617  bGotSOT = false;
618  if (bGotESC) {
619  bGotSOT = true;
620  }
621  } else {
622  bGotESC = (next_byte == ESCAPE);
623  if (bGotSOT) {
624  bGotSOT = false;
625  bInMsg = true;
626 
627  data.push_back(next_byte);
628  }
629  }
630  }
631  } // while
632 
633  return true;
634 }
