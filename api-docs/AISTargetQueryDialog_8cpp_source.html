106 void AISTargetQueryDialog::OnIdWptCreateClick(wxCommandEvent &event) {
107  if (m_MMSI != 0) { // Faulty MMSI could be reported as 0
108  auto td = g_pAIS->Get_Target_Data_From_MMSI(m_MMSI);
109  if (td) {
110  wxString n0 = wxString::Format(wxT("%s"), td->ShipName);
111  n0.Replace(_T("@"), _T(" "));
112  n0.Trim();
113  wxString mmsi = wxString::Format(wxT("%i "), td->MMSI);
114  wxString n = _T("\"") + n0 + _T("\" ") + mmsi;
115  n.append(wxDateTime::Now().Format(wxT("%H:%M")));
116  // wxString n = wxString::Format(wxT("\"%s\" %i "),td->ShipName,
117  // td->MMSI).append(wxDateTime::Now().Format(wxT("%H:%M")));
118  RoutePoint *pWP =
119  new RoutePoint(td->Lat, td->Lon, g_default_wp_icon, n, wxEmptyString);
120  pWP->m_bIsolatedMark = true; // This is an isolated mark
121  pSelect->AddSelectableRoutePoint(td->Lat, td->Lon, pWP);
122  pConfig->AddNewWayPoint(pWP, -1); // use auto next num
123 
124  if (pRouteManagerDialog && pRouteManagerDialog->IsShown())
125  pRouteManagerDialog->UpdateWptListCtrl();
126  gFrame->GetPrimaryCanvas()->undo->BeforeUndoableAction(
127  Undo_CreateWaypoint, pWP, Undo_HasParent, NULL);
128  gFrame->GetPrimaryCanvas()->undo->AfterUndoableAction(NULL);
129  Refresh(false);
130  }
131  }
132 }

134 void AISTargetQueryDialog::OnIdTrkCreateClick(wxCommandEvent &event) {
135  if (m_MMSI != 0) { // Faulty MMSI could be reported as 0
136  auto td = g_pAIS->Get_Target_Data_From_MMSI(m_MMSI);
137  if (td) {
138  if (td->b_PersistTrack) // The target was tracked and the user wants to
139  // stop it
140  {
141  td->b_PersistTrack = false;
142  g_pAIS->m_persistent_tracks.erase(td->MMSI);
143  m_createTrkBtn->SetLabel(_("Record Track"));
144  } else {
145  TrackPoint *tp = NULL;
146  TrackPoint *tp1 = NULL;
147 
148  Track *t = new Track();
149 
150  t->SetName(wxString::Format(_T("AIS %s (%u) %s %s"),
151  td->GetFullName().c_str(), td->MMSI,
152  wxDateTime::Now().FormatISODate().c_str(),
153  wxDateTime::Now().FormatISOTime().c_str()));
154  for (const AISTargetTrackPoint &ptrack_point : td->m_ptrack) {
155  vector2D point(ptrack_point.m_lon, ptrack_point.m_lat);
156  tp1 = t->AddNewPoint(point, wxDateTime(ptrack_point.m_time).ToUTC());
157  if (tp) {
158  pSelect->AddSelectableTrackSegment(tp->m_lat, tp->m_lon, tp1->m_lat,
159  tp1->m_lon, tp, tp1, t);
160  }
161  tp = tp1;
162  }
163 
164  g_TrackList.push_back(t);
165  pConfig->AddNewTrack(t);
166  // t->RebuildGUIDList(); // ensure the GUID list is
167  // intact and good
168 
169  if (pRouteManagerDialog && pRouteManagerDialog->IsShown())
170  pRouteManagerDialog->UpdateTrkListCtrl();
171  Refresh(false);
172 
173  if (wxID_YES ==
174  OCPNMessageBox(
175  NULL,
176  _("The recently captured track of this target has been "
177  "recorded.\nDo you want to continue recording until the end "
178  "of the current OpenCPN session?"),
179  _("OpenCPN Info"), wxYES_NO | wxCENTER, 60)) {
180  td->b_PersistTrack = true;
181  g_pAIS->m_persistent_tracks[td->MMSI] = t;
182  }
183  }
184  }
185  }

191  // As a display optimization....

195  // will not detract from night-vision
196 
197  long wstyle = AIS_TARGET_QUERY_STYLE;
198  if ((global_color_scheme != GLOBAL_COLOR_SCHEME_DAY) &&
199  (global_color_scheme != GLOBAL_COLOR_SCHEME_RGB))
200  wstyle |= (wxNO_BORDER);
201 
202  if (!wxFrame::Create(parent, id, caption, pos, size, wstyle)) return false;
203 
204  m_parent = parent;
205 
206  wxFont *dFont = FontMgr::Get().GetFont(_("AISTargetQuery"));
207  int font_size = wxMax(8, dFont->GetPointSize());
208  wxString face = dFont->GetFaceName();
209 #ifdef __WXGTK__
210  face = _T("Monospace");
211 #endif
212  m_basefont = FontMgr::Get().FindOrCreateFont(font_size, wxFONTFAMILY_MODERN,
213  wxFONTSTYLE_NORMAL,
214  dFont->GetWeight(), false, face);
215 
216  SetFont(*m_basefont);
217  m_adjustedFontSize = dFont->GetPointSize();
218  m_control_font_size = dFont->GetPointSize();
219 
220  CreateControls();
221 
222  SetColorScheme(global_color_scheme);
223 
224  // Set the maximum size of the entire settings dialog
225  wxSize sz = g_Platform->getDisplaySize();
226  SetSizeHints(50, 50, sz.x - 20, sz.y - 40);
227 
228  if (!m_bautosize) {
229  Fit(); // Sets the horizontal size OK

231  SetSize(-1, m_adjustedFontSize * 30); // Estimated vertical size
232  }
233

351 void AISTargetQueryDialog::AdjustBestSize(AisTargetData *td) {
352  if (!td) return;
353 
354  wxSize origSize = GetSize();
355 
356  // First pass, try to set the size using the user specified font sizes
357  // completely
358  if (!m_bsize_set) {
359  Fit();
360  RenderHTMLQuery(td);
361  m_bsize_set = true;
362  }
363 
364  int target_x = -1;
365  int target_y = -1;
366 
367  // Width adjustments
368 
369  if (m_bautosize) {
370  // Reduce the font size if necessary to eliminate horizontal scroll bars.
371  wxSize szv = m_pQueryTextCtl->GetVirtualSize();
372  if (szv.x > m_pQueryTextCtl->GetSize().x) {
373  while ((szv.x > m_pQueryTextCtl->GetSize().x) &&
374  (m_adjustedFontSize > 8)) { // fluff
375  m_adjustedFontSize--;
376 
377  RenderHTMLQuery(td);
378  m_pQueryTextCtl->Refresh();
379  m_pQueryTextCtl->Update();
380  Layout();
381  szv = m_pQueryTextCtl->GetVirtualSize();
382  }
383 
384  m_adjustedFontSize--;
385  }
386  } else {
387  wxSize szv = m_pQueryTextCtl->GetVirtualSize();
388  int csz = g_Platform->getDisplaySize().x * 8 / 10;
389  if ((szv.x) < csz) {
390  if (szv.x > m_pQueryTextCtl->GetSize().x) target_x = szv.x; // * 11/10;
391  }
392  }
393 
394  // Now adjust the font size used for the control buttons.
395  // This adjustment makes sure that the two horizontal buttons are not wider
396  // than the platform display allows.
397 
398  if (m_createWptBtn && m_createTrkBtn) {
399  wxSize psz = g_Platform->getDisplaySize();
400 
401  wxScreenDC dc;
402  wxFont *tFont = FontMgr::Get().FindOrCreateFont(
403  m_control_font_size, wxFONTFAMILY_MODERN, wxFONTSTYLE_NORMAL,
404  m_basefont->GetWeight(), false, m_basefont->GetFaceName());
405  dc.SetFont(*tFont);
406 
407  wxSize tsz = dc.GetTextExtent(m_createWptBtn->GetLabel() +
408  m_createTrkBtn->GetLabel());
409 
410  float totalButtonWidth = tsz.x;
411 
412  if (totalButtonWidth * 1.5 > psz.x) {
413  float delta = (float)totalButtonWidth * 2. / psz.x;
414 
415  float font_size = m_control_font_size / delta;
416 
417  wxFont *fp_font = FontMgr::Get().FindOrCreateFont(
418  font_size, wxFONTFAMILY_MODERN, wxFONTSTYLE_NORMAL,
419  m_basefont->GetWeight(), false, m_basefont->GetFaceName());
420 
421  m_createWptBtn->SetFont(*fp_font);
422  m_createTrkBtn->SetFont(*fp_font);
423 
424  m_control_font_size = font_size;
425  }
426  }
427 
428  // Height adjustments
429  // Try to avoid vertical scroll bar if possible.
430 
431  // Estimate the control button area height
432  int yb = 0;
433  if (m_createWptBtn) yb = m_createWptBtn->GetSize().y * 4;
434 
435  wxSize szyv = m_pQueryTextCtl->GetVirtualSize();
436  int csz = g_Platform->getDisplaySize().y * 85 / 100;
437  if ((szyv.y + yb) < csz) {
438  if (szyv.y > m_pQueryTextCtl->GetSize().y)
439  target_y = (szyv.y * 11 / 10) + yb;
440  } else { // Probably going to be a vertical scroll bar, so adjust width
441  // slightly
442  target_y = csz;
443  target_x = szyv.x * 11 / 10;
444  }
445 
446  SetSize(target_x, target_y);
447 
448  wxSize nowSize = GetSize();
449 
450  if (nowSize != origSize) {
451  if (m_bautoCentre) Centre();
452  }
453 }

455 void AISTargetQueryDialog::RenderHTMLQuery(AisTargetData *td) {
456  int font_size = m_adjustedFontSize;
457  wxFont *fp_font = FontMgr::Get().FindOrCreateFont(
458  font_size, wxFONTFAMILY_MODERN, wxFONTSTYLE_NORMAL,
459  m_basefont->GetWeight(), false, m_basefont->GetFaceName());
460 
461  SetFont(*fp_font);
462 
463  int sizes[7];
464  for (int i = -2; i < 5; i++) {
465  sizes[i + 2] = fp_font->GetPointSize() + i + (i > 0 ? i : 0);
466  }
467 
468  wxString html;
469  wxColor bg = GetBackgroundColour();
470  wxColor fg = GetForegroundColour();
471 
472  html.Printf(
473  _T("<html><body bgcolor=#%02x%02x%02x><font ")
474  _T("color=#%02x%02x%02x><center>"),
475  bg.Red(), bg.Green(), bg.Blue(), fg.Red(), fg.Green(), fg.Blue());
476 
477  html << td->BuildQueryResult();
478 
479  html << _T("</center></font></body></html>");
480 
481  m_pQueryTextCtl->SetFonts(fp_font->GetFaceName(), fp_font->GetFaceName(),
482  sizes);
483 
484  wxCharBuffer buf = html.ToUTF8();
485  if (buf.data()) // string OK?
486  m_pQueryTextCtl->SetPage(html);
