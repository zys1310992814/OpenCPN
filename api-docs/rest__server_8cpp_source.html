118 static void HandleRxObject(struct mg_connection* c, struct mg_http_message* hm,
119  RestServer* parent) {
120  int MID = ORS_CHUNK_N;
121 
122  std::string api_key = HttpVarToString(hm->query, "apikey");
123  std::string source = HttpVarToString(hm->query, "source");
124  std::string force = HttpVarToString(hm->query, "force");
125  std::string activate = HttpVarToString(hm->query, "activate");
126  std::string xml_content;
127  if (hm->chunk.len)
128  xml_content = std::string(hm->chunk.ptr, hm->chunk.len);
129  else {
130  MID = ORS_CHUNK_LAST;
131  }
132  mg_http_delete_chunk(c, hm);
133  parent->UpdateReturnStatus(RestServerResult::Void);
134 
135  if (!source.empty()) {
136  assert(parent && "Null parent pointer");
137  auto data_ptr =
138  std::make_shared<RestIoEvtData>(RestIoEvtData::CreateCmdData(
139  api_key, source, xml_content, !force.empty(), !activate.empty()));
140  PostEvent(parent, data_ptr, MID);
141  }
142  if (MID == ORS_CHUNK_LAST) {
143  std::unique_lock<std::mutex> lock{parent->ret_mutex};
144  bool r = parent->return_status_cv.wait_for(lock, 10s, [&] {
145  return parent->GetReturnStatus() != RestServerResult::Void;
146  });
147  if (!r) wxLogWarning("Timeout waiting for REST server condition");
148  mg_http_reply(c, 200, "", "{\"result\": %d}\n", parent->GetReturnStatus());
149  }
150 }

154  std::string api_key = HttpVarToString(hm->query, "apikey");

168  mg_http_reply(c, 200, "", "{\"result\": %d, \"version\": \"%s\"}\n",

192 // We use the same event handler function for HTTP and HTTPS connections
193 // fn_data is NULL for plain HTTP, and non-NULL for HTTPS
194 static void fn(struct mg_connection* c, int ev, void* ev_data, void* fn_data) {
195  auto parent = static_cast<RestServer*>(fn_data);
196 
197  if (ev == MG_EV_ACCEPT /*&& fn_data != NULL*/) {
198  struct mg_tls_opts opts = {0};
199  opts.ca = nullptr; // "cert.pem" Uncomment to enable two-way SSL
200  opts.cert = parent->m_cert_file.c_str();
201  opts.certkey = parent->m_key_file.c_str();
202  opts.ciphers = nullptr;
203  mg_tls_init(c, &opts);
204  } else if (ev == MG_EV_TLS_HS) { // Think of this as "start of session"
205  PostEvent(parent, nullptr, ORS_START_OF_SESSION);
206  } else if (ev == MG_EV_HTTP_CHUNK) {
207  auto hm = (struct mg_http_message*)ev_data;
208  if (mg_http_match_uri(hm, "/api/ping")) {
209  HandlePing(c, hm, parent);
210  } else if (mg_http_match_uri(hm, "/api/rx_object")) {
211  HandleRxObject(c, hm, parent);
212  } else if (mg_http_match_uri(hm, "/api/writable")) {
213  HandleWritable(c, hm, parent);
214  } else if (mg_http_match_uri(hm, "/api/get-version")) {
215  std::string reply(kVersionReply);
216  ocpn::replace(reply, "@version@", PACKAGE_VERSION);
217  mg_http_reply(c, 200, "", reply.c_str());
218  } else {
219  mg_http_reply(c, 404, "", "url: not found");
220  }
221  }
222 }

254 void RestServer::IoThread::Run() {
255  run_flag = 1;
256  struct mg_mgr mgr = {0}; // Event manager
257  mg_log_set(MG_LL_DEBUG); // Set log level
258  mg_mgr_init(&mgr); // Initialise event manager
259 
260  // Create HTTPS listener
261  MESSAGE_LOG << "Listening on " << m_server_ip << "\n";
262  mg_http_listen(&mgr, m_server_ip.c_str(), fn, &m_parent);
263 
264  while (run_flag > 0) mg_mgr_poll(&mgr, 200); // Infinite event loop
265  mg_mgr_free(&mgr);
266  run_flag = -1;
267  m_parent.m_exit_sem.Post();
268 }

364 bool RestServer::CheckApiKey(const RestIoEvtData& evt_data) {
365  // Look up the api key in the hash map. If found, we are done.
366  if (m_key_map.find(evt_data.source) != m_key_map.end()) {
367  if (m_key_map[evt_data.source] == evt_data.api_key) return true;
368  }
369  // Need a new PIN confirmation, add it to map and persist
370  m_pincode = Pincode::Create();
371  std::string new_api_key = m_pincode.Hash();
372  if (evt_data.api_key.size() < 10) // client sends old-style keys
373  new_api_key = m_pincode.CompatHash();
374 
375  m_key_map[evt_data.source] = new_api_key;
376  SaveConfig();
377 
378  std::stringstream ss;
379  ss << evt_data.source << " " << _("wants to send you new data.") << "\n"
380  << _("Please enter the following PIN number on ") << evt_data.source << " "
381  << _("to pair with this device") << "\n";
382  if (m_pin_dialog) m_pin_dialog->Destroy();
383  m_pin_dialog = m_dlg_ctx.run_pincode_dlg(ss.str(), m_pincode.ToString());
384 
385  return false;
386 }

388 void RestServer::HandleServerMessage(ObservedEvt& event) {
389  auto evt_data = UnpackEvtPointer<RestIoEvtData>(event);
390  switch (event.GetId()) {
391  case ORS_START_OF_SESSION:
392  // Prepare a temp file to catch chuncks that might follow
393  m_upload_path = wxFileName::CreateTempFileName("ocpn_tul").ToStdString();
394 
395  m_ul_stream.open(m_upload_path.c_str(), std::ios::out | std::ios::trunc);
396  if (!m_ul_stream.is_open()) {
397  wxLogMessage("REST_server: Cannot open %s for write", m_upload_path);
398  m_upload_path.clear(); // reset for next time.
399  return;
400  }
401  return;
402  case ORS_CHUNK_N:
403  // Stream out to temp file
404  if (!m_upload_path.empty() && m_ul_stream.is_open()) {
405  m_ul_stream.write(evt_data->payload.c_str(), evt_data->payload.size());
406  }
407  return;
408  case ORS_CHUNK_LAST:
409  // Cancel existing dialog and close temp file
410  if (m_pin_dialog) wxQueueEvent(m_pin_dialog, new wxCloseEvent);
411  if (!m_upload_path.empty() && m_ul_stream.is_open()) m_ul_stream.close();
412  break;
413  }
414 
415  if (!CheckApiKey(*evt_data)) {
416  UpdateReturnStatus(RestServerResult::NewPinRequested);
417  return;
418  }
419 
420  UpdateReturnStatus(RestServerResult::NoError);
421  switch (evt_data->cmd) {
422  case RestIoEvtData::Cmd::Ping:
423  return;
424  case RestIoEvtData::Cmd::CheckWrite: {
425  auto guid = evt_data->payload;
426  auto dup = m_route_ctx.find_route_by_guid(guid);
427  if (!dup || evt_data->force || m_overwrite) {
428  UpdateReturnStatus(RestServerResult::NoError);
429  } else {
430  UpdateReturnStatus(RestServerResult::DuplicateRejected);
431  }
432  return;
433  }
434  case RestIoEvtData::Cmd::Object: {
435  pugi::xml_document doc;
436  pugi::xml_parse_result result = doc.load_file(m_upload_path.c_str());
437  if (result.status == pugi::status_ok) {
438  m_upload_path.clear(); // empty for next time
439 
440  pugi::xml_node objects = doc.child("gpx");
441  for (pugi::xml_node object = objects.first_child(); object;
442  object = object.next_sibling()) {
443  if (!strcmp(object.name(), "rte")) {
444  HandleRoute(object, *evt_data);
445  } else if (!strcmp(object.name(), "trk")) {
446  HandleTrack(object, *evt_data);
447  } else if (!strcmp(object.name(), "wpt")) {
448  HandleWaypoint(object, *evt_data);
449  }
450  }
451  } else {
452  UpdateReturnStatus(RestServerResult::ObjectParseError);
453  }
454  break;
455  }
456  }
457 }

459 void RestServer::HandleRoute(pugi::xml_node object,
460  const RestIoEvtData& evt_data) {
461  Route* route = GPXLoadRoute1(object, true, false, false, 0, true);
462  // Check for duplicate GUID
463  bool add = true;
464  bool overwrite_one = false;
465  Route* duplicate = m_route_ctx.find_route_by_guid(route->GetGUID());
466  if (duplicate) {
467  if (!m_overwrite && !evt_data.force) {
468  auto result = m_dlg_ctx.run_accept_object_dlg(
469  _("The received route already exists on this system.\nReplace?"),
470  _("Always replace objects?"));
471  if (result.status != ID_STG_OK) {
472  add = false;
473  UpdateReturnStatus(RestServerResult::DuplicateRejected);
474  } else {
475  m_overwrite = result.check1_value;
476  overwrite_one = true;
477  SaveConfig();
478  }
479  }
480  }
481  if (add) {
482  if (m_overwrite || overwrite_one || evt_data.force) {
483  // Remove the existing duplicate route before adding new route
484  m_route_ctx.delete_route(duplicate);
485  }
486  // Add the route to the global list
487  NavObjectCollection1 pSet;
488  if (InsertRouteA(route, &pSet)) {
489  UpdateReturnStatus(RestServerResult::NoError);
490  if (evt_data.activate)
491  activate_route.Notify(route->GetGUID().ToStdString());
492  if (g_pRouteMan) g_pRouteMan->on_routes_update.Notify();
493  } else {
494  UpdateReturnStatus(RestServerResult::RouteInsertError);
495  }
496  }
497  UpdateRouteMgr();
498 }
