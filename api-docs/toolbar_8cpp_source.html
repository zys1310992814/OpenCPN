398 int ocpnFloatingToolbarDialog::RebuildToolbar() {
399  ocpnToolBarSimple *tb = GetToolbar();
400  if (!tb) return 0;
401 
402  // Iterate over the array of items added,
403  // Creating the toolbar from enabled items.
404  int i_count = 0;
405  for (auto it = m_Items.cbegin(); it != m_Items.cend(); it++) {
406  ToolbarItemContainer *tic = *it;
407  if (!tic) continue;
408 
409  bool bEnabled = _toolbarConfigMenuUtil(tic);
410 
411  if (bEnabled) {
412  wxToolBarToolBase *tool =
413  tb->AddTool(tic->m_ID, tic->m_label, tic->m_bmpNormal,
414  tic->m_bmpDisabled, tic->m_toolKind, tic->m_tipString);
415  tic->m_tool = tool;
416 
417  // Plugin tools may have prescribed their own SVG toolbars as file
418  // locations.
419  if (!tic->m_NormalIconSVG.IsEmpty()) {
420  tb->SetToolBitmapsSVG(tic->m_ID, tic->m_NormalIconSVG,
421  tic->m_RolloverIconSVG, tic->m_ToggledIconSVG);
422  }
423  }
424 
425  i_count++;
426  }
427 
428  return i_count;
429 }

470 bool ocpnFloatingToolbarDialog::_toolbarConfigMenuUtil(
471  ToolbarItemContainer *tic) {
472  if (m_FloatingToolbarConfigMenu) {
473  wxMenuItem *menuitem;
474 
475  if (tic->m_ID == ID_MOB && g_bPermanentMOBIcon) return true;
476 
477  if (tic->m_bRequired) return true;
478  if (tic->m_bPlugin) return true;
479 
480  // Item ID trickery is needed because the wxCommandEvents for menu item
481  // clicked and toolbar button clicked are 100% identical, so if we use same
482  // id's we can't tell the events apart.
483 
484  int idOffset = 100; // Hopefully no more than 100 total icons...
485  int menuItemId = tic->m_ID + idOffset;
486 
487  menuitem = m_FloatingToolbarConfigMenu->FindItem(menuItemId);
488 
489  if (menuitem) {
490  return menuitem->IsChecked();
491  }
492 
493  menuitem = m_FloatingToolbarConfigMenu->AppendCheckItem(menuItemId,
494  tic->m_tipString);
495  size_t n = m_FloatingToolbarConfigMenu->GetMenuItemCount();
496  menuitem->Check(m_configString.Len() >= n
497  ? m_configString.GetChar(n - 1) == _T('X')
498  : true);
499  return menuitem->IsChecked();
500  } else
501  return true;
502 }

643 void ocpnFloatingToolbarDialog::SetDefaultPosition() {
644  if (m_block) return;
645 
646  if (m_pparent && m_ptoolbar) {
647  wxSize cs = m_pparent->GetClientSize();
648  if (-1 == m_dock_x)
649  m_position.x = m_dock_min_x;
650  else if (1 == m_dock_x)
651  m_position.x = cs.x - GetSize().x;
652 
653  if (-1 == m_dock_y)
654  m_position.y = m_dock_min_y;
655  else if (1 == m_dock_y)
656  m_position.y = cs.y - GetSize().y;
657 
658  m_position.x = wxMin(cs.x - GetSize().x, m_position.x);
659  m_position.y = wxMin(cs.y - GetSize().y, m_position.y);
660 
661  m_position.x = wxMax(m_dock_min_x, m_position.x);
662  m_position.y = wxMax(m_dock_min_y, m_position.y);
663 
664  m_position.y += m_auxOffsetY;

667  g_maintoolbar_y = m_position.y;

1019  } else {

1021 
1022  if (i == 1) {
1023  barsize.x += m_style->GetIcon(_T("grabber")).GetWidth();
1024  }
1025  if (i == lines && i > 1) {
1026  int toolsInLastLine = toolCount % m_ptoolbar->GetMaxCols();

1028  toolsInLastLine = m_ptoolbar->GetMaxCols();
