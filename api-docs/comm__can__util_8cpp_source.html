45 unsigned long BuildCanID(int priority, int source, int destination, int pgn) {
46  // build CanID
47  unsigned long cid = 0;
48  unsigned char pf = (unsigned char) (pgn >> 8);
49  if (pf < 240){
50  cid = ((unsigned long)(priority & 0x7))<<26 | pgn<<8 | ((unsigned long)destination)<<8 | (unsigned long)source;
51  }
52  else {
53  cid = ((unsigned long)(priority & 0x7))<<26 | pgn<<8 | (unsigned long)source;
54  }
55  return cid;
56 }

61 CanHeader::CanHeader(const CanFrame frame) {
62  unsigned char buf[4];
63  buf[0] = frame.can_id & 0xFF;
64  buf[1] = (frame.can_id >> 8) & 0xFF;
65  buf[2] = (frame.can_id >> 16) & 0xFF;
66  buf[3] = (frame.can_id >> 24) & 0xFF;
67 
68  source = buf[0];
69  destination = buf[2] < 240 ? buf[1] : 255;
70  pgn = (buf[3] & 0x01) << 16 | (buf[2] << 8) | (buf[2] < 240 ? 0 : buf[1]);
71  priority = (buf[3] & 0x1c) >> 2;
72 }

105 int FastMessageMap::FindMatchingEntry(const CanHeader header,
106  const unsigned char sid) {
107  for (unsigned i = 0; i < entries.size(); i++) {
108  if (((sid & 0xE0) == (entries[i].sid & 0xE0)) &&
109  (entries[i].header.pgn == header.pgn) &&
110  (entries[i].header.source == header.source) &&
111  (entries[i].header.destination == header.destination)) {
112  return i;
113  }
114  }
115  return kNotFound;

133 bool FastMessageMap::InsertEntry(const CanHeader header,
134  const unsigned char* data, int index) {
135  // first message of fast packet
136  // data[0] Sequence Identifier (sid)
137  // data[1] Length of data bytes
138  // data[2..7] 6 data bytes
139 
140  CheckGc();
141  // Ensure that this is indeed the first frame of a fast message
142  if ((data[0] & 0x1F) == 0) {
143  int total_data_len; // will also include padding as we memcpy all of the
144  // frame, because I'm lazy
145  total_data_len = static_cast<unsigned int>(data[1]);
146  total_data_len += 7 - ((total_data_len - 6) % 7);
147 
148  entries[index].sid = static_cast<unsigned int>(data[0]);
149  entries[index].expected_length = static_cast<unsigned int>(data[1]);
150  entries[index].header = header;
151  entries[index].time_arrived = wxDateTime::Now();
152 
153  entries[index].data.resize(total_data_len);
154  memcpy(&entries[index].data[0], &data[2], 6);
155  // First frame of a multi-frame Fast Message contains six data bytes.
156  // Position the cursor ready for next message
157  entries[index].cursor = 6;
158 
159  // Fusion, using fast messages to sends frames less than eight bytes
160  return entries[index].expected_length <= 6;
161  }
162  return false;
163  // No further processing is performed if this is not a start frame.
164  // A start frame may have been dropped and we received a subsequent frame
165 }

167 bool FastMessageMap::AppendEntry(const CanHeader header,
168  const unsigned char* data, int position) {
169  // Check that this is the next message in the sequence
170  if ((entries[position].sid + 1) == data[0]) {
171  memcpy(&entries[position].data[entries[position].cursor], &data[1], 7);
172  entries[position].sid = data[0];
173  // Subsequent messages contains seven data bytes (last message may be padded
174  // with 0xFF)
175  entries[position].cursor += 7;
176  // Is this the last message ?
177  return entries[position].cursor >= entries[position].expected_length;
178  } else if ((data[0] & 0x1F) == 0) {
179  // We've found a matching entry, however this is a start frame, therefore
180  // we've missed an end frame, and now we have a start frame with the same id
181  // (top 3 bits). The id has obviously rolled over. Should really double
182  // check that (data[0] & 0xE0) Clear the entry as we don't want to leak
183  // memory, prior to inserting a start frame
184  entries.erase(entries.begin() + position);
185  position = AddNewEntry();
186  // And now insert it
187  InsertEntry(header, data, position);
188  // FIXME (dave) Should update the dropped frame stats
189  return true;
190  } else {
191  // This is not the next frame in the sequence and not a start frame
192  // We've dropped an intermedite frame, so free the slot and do no further
193  // processing
194  entries.erase(entries.begin() + position);
195  // Dropped Frame Statistics
196  if (dropped_frames == 0) {
197  dropped_frame_time = wxDateTime::Now();
198  dropped_frames += 1;
199  } else {
200  dropped_frames += 1;
201  }
202  // FIXME (dave)
203  // if ((dropped_frames > CONST_DROPPEDFRAME_THRESHOLD) &&
204  // (wxDateTime::Now() < (dropped_frame_time +
205  // wxTimeSpan::Seconds(CONST_DROPPEDFRAME_PERIOD) ) ) ) {
206  // wxLogError(_T("TwoCan Device, Dropped Frames rate exceeded"));

208  // Priority: %d PGN: %d"),header.source, header.destination,
