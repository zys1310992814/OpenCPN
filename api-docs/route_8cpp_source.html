106 void Route::CloneRoute(Route *psourceroute, int start_nPoint, int end_nPoint,
107  const wxString &suffix,
108  const bool duplicate_first_point) {
109  m_RouteNameString = psourceroute->m_RouteNameString + suffix;
110  m_RouteStartString = psourceroute->m_RouteStartString;
111  m_RouteEndString = psourceroute->m_RouteEndString;
112 
113  int i;
114  for (i = start_nPoint; i <= end_nPoint; i++) {
115  if (!psourceroute->m_bIsInLayer &&
116  !(i == start_nPoint && duplicate_first_point)) {
117  AddPoint(psourceroute->GetPoint(i), false);
118  }
119  else {
120  RoutePoint *psourcepoint = psourceroute->GetPoint(i);
121  RoutePoint *ptargetpoint = new RoutePoint(

124  ptargetpoint->m_bShowName =
125  psourcepoint->m_bShowName; // do not change new wpt's name visibility
126  AddPoint(ptargetpoint, false);
127  }
128  }
129 
130  FinalizeForRendering();

238 static void TestLongitude(double lon, double min, double max, bool &lonl,
239  bool &lonr) {
240  double clon = (min + max) / 2;
241  if (min - lon > 180) lon += 360;
242 
243  lonl = lonr = false;
244  if (lon < min) {
245  if (lon < clon - 180)
246  lonr = true;
247  else
248  lonl = true;
249  } else if (lon > max) {
250  if (lon > clon + 180)
251  lonl = true;
252  else
253  lonr = true;
254  }
255 }

426 LLBBox &Route::GetBBox(void) {
427  if (RBBox.GetValid()) return RBBox;
428 
429  double bbox_lonmin, bbox_lonmax, bbox_latmin, bbox_latmax;
430 
431  wxRoutePointListNode *node = pRoutePointList->GetFirst();
432  RoutePoint *data = node->GetData();
433 
434  if (data->m_wpBBox.GetValid()) {
435  bbox_lonmax = data->m_wpBBox.GetMaxLon();
436  bbox_lonmin = data->m_wpBBox.GetMinLon();
437  bbox_latmax = data->m_wpBBox.GetMaxLat();
438  bbox_latmin = data->m_wpBBox.GetMinLat();
439  } else {
440  bbox_lonmax = bbox_lonmin = data->m_lon;
441  bbox_latmax = bbox_latmin = data->m_lat;
442  }
443 
444  double lastlon = data->m_lon, wrap = 0;
445 
446  node = node->GetNext();
447  while (node) {
448  data = node->GetData();
449 
450  if (lastlon - data->m_lon > 180)
451  wrap += 360;
452  else if (data->m_lon - lastlon > 180)
453  wrap -= 360;
454 
455  double lon = data->m_lon + wrap;
456 
457  if (lon > bbox_lonmax) bbox_lonmax = lon;
458  if (lon < bbox_lonmin) bbox_lonmin = lon;
459 
460  if (data->m_lat > bbox_latmax) bbox_latmax = data->m_lat;
461  if (data->m_lat < bbox_latmin) bbox_latmin = data->m_lat;
462 
463  lastlon = data->m_lon;
464  node = node->GetNext();
465  }
466 
467  if (bbox_lonmin < -360)
468  bbox_lonmin += 360, bbox_lonmax += 360;
469  else if (bbox_lonmax > 360)
470  bbox_lonmin -= 360, bbox_lonmax -= 360;
471 
472  if (bbox_lonmax - bbox_lonmin > 360) bbox_lonmin = -180, bbox_lonmax = 180;
473 
474  RBBox.Set(bbox_latmin, bbox_lonmin, bbox_latmax, bbox_lonmax);
475 
476  return RBBox;
477 }

479 /*
480  Update a single route segment lengths
481  Also, compute total route length by summing segment distances.
482  */
483 void Route::UpdateSegmentDistance(RoutePoint *prp0, RoutePoint *prp,
484  double planspeed) {
485  double slat1 = prp0->m_lat, slon1 = prp0->m_lon;
486  double slat2 = prp->m_lat, slon2 = prp->m_lon;
487 
488  // Calculate the absolute distance from 1->2
489 
490  double dd;
491  double br;
492  // why are we using mercator rather than great circle here?? [sean 8-11-2015]
493  DistanceBearingMercator(slat2, slon2, slat1, slon1, &br, &dd);
494 
495  prp->SetCourse(br);
496  prp->SetDistance(dd);
497 
498  // And store in Point 2
499  prp->m_seg_len = dd;
500 
501  m_route_length += dd;
502 
503  // If Point1 Description contains VMG, store it for Properties Dialog in
504  // Point2 If Point1 Description contains ETD, store it in Point1
505 
506  if (planspeed > 0.) {
507  wxDateTime etd;
508 
509  double legspeed = planspeed;
510  if (prp->GetPlannedSpeed() > 0.1 && prp->GetPlannedSpeed() < 1000.)
511  legspeed = prp->GetPlannedSpeed();
512  if (legspeed > 0.1 && legspeed < 1000.) {
513  m_route_time += 3600. * dd / legspeed;
514  prp->m_seg_vmg = legspeed;
515  }
516  wxLongLong duration = wxLongLong(3600.0 * prp->m_seg_len / prp->m_seg_vmg);
517  prp->SetETE(duration);
518  wxTimeSpan ts(0, 0, duration);
519  if (!prp0->GetManualETD().IsValid()) {
520  prp0->m_manual_etd = false;
521  if (prp0->GetETA().IsValid()) {
522  prp0->m_seg_etd = prp0->GetETA();
523  } else {
524  prp0->m_seg_etd =
525  m_PlannedDeparture + wxTimeSpan(0, 0, m_route_time - duration);
526  }
527  }
528 
529  prp->m_seg_eta = prp0->GetETD() + ts;
530  if (!prp->m_manual_etd || !prp->GetETD().IsValid()) {
531  prp->m_seg_etd = prp->m_seg_eta;
532  prp->m_manual_etd = false;
533  }
534  }
535 }

537 /*
538  Update the route segment lengths, storing each segment length in <destination>
539  point. Also, compute total route length by summing segment distances.
540  */
541 void Route::UpdateSegmentDistances(double planspeed) {
542  wxPoint rpt, rptn;
543 
544  m_route_length = 0.0;
545  m_route_time = 0.0;
546 
547  wxRoutePointListNode *node = pRoutePointList->GetFirst();
548 
549  if (node) {
550  // Route start point
551  RoutePoint *prp0 = node->GetData();
552  if (!prp0->m_manual_etd) {
553  prp0->m_seg_eta = m_PlannedDeparture;
554  prp0->m_seg_etd = m_PlannedDeparture;
555  }
556  node = node->GetNext();
557 
558  while (node) {
559  RoutePoint *prp = node->GetData();
560  UpdateSegmentDistance(prp0, prp, planspeed);
561 
562  prp0 = prp;
563 
564  node = node->GetNext();
565  }
566  }

675 // Is this route equal to another, meaning,
676 // Do all routepoint positions and names match?
677 bool Route::IsEqualTo(Route *ptargetroute) {
678  wxRoutePointListNode *pthisnode = (this->pRoutePointList)->GetFirst();
679  wxRoutePointListNode *pthatnode = (ptargetroute->pRoutePointList)->GetFirst();
680 
681  if (NULL == pthisnode) return false;
682 
683  if (this->m_bIsInLayer || ptargetroute->m_bIsInLayer) return false;
684 
685  if (this->GetnPoints() != ptargetroute->GetnPoints()) return false;
686 
687  while (pthisnode) {
688  if (NULL == pthatnode) return false;
689 
690  RoutePoint *pthisrp = pthisnode->GetData();
691  RoutePoint *pthatrp = pthatnode->GetData();
