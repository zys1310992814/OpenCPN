354 void CanvasOptions::RefreshControlValues(void) {
355  ChartCanvas* parentCanvas = wxDynamicCast(m_parent, ChartCanvas);
356  if (!parentCanvas) return;
357 
358  // Control options
359  // pCBToolbar->SetValue(parentCanvas->GetToolbarEnable());
360 
361  // Navigation Mode
362  int mode = parentCanvas->GetUpMode();
363  if (mode == NORTH_UP_MODE)
364  pCBNorthUp->SetValue(true);
365  else if (mode == COURSE_UP_MODE)
366  pCBCourseUp->SetValue(true);
367  else
368  pCBHeadUp->SetValue(true);
369 
370  pCBLookAhead->SetValue(parentCanvas->GetLookahead());
371 
372  // Display options
373  pCDOQuilting->SetValue(parentCanvas->GetQuiltMode());
374  pSDisplayGrid->SetValue(parentCanvas->GetShowGrid());
375  pCDOOutlines->SetValue(parentCanvas->GetShowOutlines());
376  pSDepthUnits->SetValue(parentCanvas->GetShowDepthUnits());
377 
378  // AIS Options
379  pCBShowAIS->SetValue(parentCanvas->GetShowAIS());
380  pCBAttenAIS->SetValue(parentCanvas->GetAttenAIS());
381 
382  // Tide/Current
383  pCDOTides->SetValue(parentCanvas->GetbShowTide());
384  pCDOCurrents->SetValue(parentCanvas->GetbShowCurrent());
385  ;
386 
387  // ENC Options
388  pCDOENCText->SetValue(parentCanvas->GetShowENCText());
389  pCBENCDepth->SetValue(parentCanvas->GetShowENCDepth());
390  pCBENCLightDesc->SetValue(parentCanvas->GetShowENCLightDesc());
391  pCBENCBuoyLabels->SetValue(parentCanvas->GetShowENCBuoyLabels());
392  pCBENCLights->SetValue(parentCanvas->GetShowENCLights());
393  pCBENCAnchorDetails->SetValue(parentCanvas->GetShowENCAnchor());
394  pCBENCVisibleSectors->SetValue(parentCanvas->GetShowVisibleSectors());
395  pCBENCDataQuality->SetValue(parentCanvas->GetShowENCDataQual());
396 
397  // pCBENCLightDesc->Enable(parentCanvas->GetShowENCLights());
398 
399  // Display category
400  int nset = 2; // default OTHER
401  switch (parentCanvas->GetENCDisplayCategory()) {
402  case (DISPLAYBASE):
403  nset = 0;
404  break;
405  case (STANDARD):
406  nset = 1;
407  break;
408  case (OTHER):
409  nset = 2;
410  break;
411  case (MARINERS_STANDARD):
412  nset = 3;
413  break;
414  default:
415  nset = 3;
416  break;
417  }
418  m_pDispCat->SetSelection(nset);
419 
420  // If no ENCs are available in the current canvas group, then disable the ENC
421  // related options.
422  pCDOENCText->Enable(m_ENCAvail);
423  pCBENCDepth->Enable(m_ENCAvail);
424  pCBENCLightDesc->Enable(m_ENCAvail && parentCanvas->GetShowENCLights());
425  pCBENCBuoyLabels->Enable(m_ENCAvail);
426  pCBENCLights->Enable(m_ENCAvail);
427  pCBENCVisibleSectors->Enable(m_ENCAvail);
428 
429  // Anchor conditions and dateQuality are only available if display category
430  // is "All" or "User Standard"
431  pCBENCDataQuality->Enable(m_ENCAvail && (nset > 1));
432  pCBENCAnchorDetails->Enable(m_ENCAvail && (nset > 1));
433 
434  // Many options are not valid if display category is "Base"
435  if (nset == 0) {
436  pCDOENCText->Disable();
437  pCBENCDepth->Disable();
438  pCBENCLightDesc->Disable();
439  pCBENCBuoyLabels->Disable();
440  pCBENCLights->Disable();
441  pCBENCVisibleSectors->Disable();
442  pCBENCDataQuality->Disable();
443  }
444 
445  m_pDispCat->Enable(m_ENCAvail);
446 }

453 void CanvasOptions::UpdateCanvasOptions(void) {
454  ChartCanvas* parentCanvas = wxDynamicCast(m_parent, ChartCanvas);
455  if (!parentCanvas) return;
456 
457  bool b_needRefresh = false;
458  bool b_needReLoad = false;
459 
460  // if(pCBToolbar->GetValue() != parentCanvas->GetToolbarEnable()){
461  // parentCanvas->SetToolbarEnable( pCBToolbar->GetValue() );
462  // b_needRefresh = true;
463  // }
464 
465  if (pCDOQuilting->GetValue() != parentCanvas->GetQuiltMode()) {
466  parentCanvas->ToggleCanvasQuiltMode();
467  }
468 
469  if (pSDisplayGrid->GetValue() != parentCanvas->GetShowGrid()) {
470  parentCanvas->SetShowGrid(pSDisplayGrid->GetValue());
471  b_needRefresh = true;
472  }
473 
474  if (pCDOOutlines->GetValue() != parentCanvas->GetShowOutlines()) {
475  parentCanvas->SetShowOutlines(pCDOOutlines->GetValue());
476  b_needRefresh = true;
477  }
478  if (pSDepthUnits->GetValue() != parentCanvas->GetShowDepthUnits()) {
479  parentCanvas->SetShowDepthUnits(pSDepthUnits->GetValue());
480  b_needRefresh = true;
481  }
482 
483  if (pCBShowAIS->GetValue() != parentCanvas->GetShowAIS()) {
484  parentCanvas->SetShowAIS(pCBShowAIS->GetValue());
485  b_needRefresh = true;
486  }
487 
488  if (pCBAttenAIS->GetValue() != parentCanvas->GetAttenAIS()) {
489  parentCanvas->SetAttenAIS(pCBAttenAIS->GetValue());
490  b_needRefresh = true;
491  }
492 
493  if (pCDOTides->GetValue() != parentCanvas->GetbShowTide()) {
494  parentCanvas->ShowTides(pCDOTides->GetValue());
495  b_needRefresh = true;
496  }
497  if (pCDOCurrents->GetValue() != parentCanvas->GetbShowCurrent()) {
498  parentCanvas->ShowCurrents(pCDOCurrents->GetValue());
499  b_needRefresh = true;
500  }
501 
502  // ENC Options
503  if (pCDOENCText->GetValue() != parentCanvas->GetShowENCText()) {
504  parentCanvas->SetShowENCText(pCDOENCText->GetValue());
505  b_needReLoad = true;
506  }
507 
508  if (pCBENCDepth->GetValue() != parentCanvas->GetShowENCDepth()) {
509  parentCanvas->SetShowENCDepth(pCBENCDepth->GetValue());
510  b_needReLoad = true;
511  }
512 
513  if (pCBENCLightDesc->GetValue() != parentCanvas->GetShowENCLightDesc()) {
514  parentCanvas->SetShowENCLightDesc(pCBENCLightDesc->GetValue());
515  b_needReLoad = true;
516  }
517 
518  if (pCBENCBuoyLabels->GetValue() != parentCanvas->GetShowENCBuoyLabels()) {
519  parentCanvas->SetShowENCBuoyLabels(pCBENCBuoyLabels->GetValue());
520  b_needReLoad = true;
521  }
522 
523  if (pCBENCLights->GetValue() != parentCanvas->GetShowENCLights()) {
524  parentCanvas->SetShowENCLights(pCBENCLights->GetValue());
525  b_needReLoad = true;
526  }
527 
528  if (pCBENCAnchorDetails->GetValue() != parentCanvas->GetShowENCAnchor()) {
529  parentCanvas->SetShowENCAnchor(pCBENCAnchorDetails->GetValue());
530  b_needReLoad = true;
531  }
532 
533  if (pCBENCVisibleSectors->GetValue() !=
534  parentCanvas->GetShowVisibleSectors()) {
535  parentCanvas->SetShowVisibleSectors(pCBENCVisibleSectors->GetValue());
536  b_needReLoad = true;
537  }
538 
539  if (pCBENCDataQuality->GetValue() !=
540  parentCanvas->GetShowENCDataQual()) {
541  parentCanvas->SetShowENCDataQual(pCBENCDataQuality->GetValue());
542  b_needReLoad = true;
543  }
544 
545  int newMode = NORTH_UP_MODE;
546  if (pCBCourseUp->GetValue())
547  newMode = COURSE_UP_MODE;
548  else if (pCBHeadUp->GetValue())
549  newMode = HEAD_UP_MODE;
550 
551  if (newMode != parentCanvas->GetUpMode()) {
552  parentCanvas->SetUpMode(newMode);
553  b_needReLoad = true;
554  }
555 
556  if (pCBLookAhead->GetValue() != parentCanvas->GetLookahead()) {
557  parentCanvas->ToggleLookahead();
558  parentCanvas->UpdateFollowButtonState();
559  b_needReLoad = true;
560  }
561 
562  int nset = 2;
563  switch (parentCanvas->GetENCDisplayCategory()) {
564  case (DISPLAYBASE):
565  nset = 0;
566  break;
567  case (STANDARD):
568  nset = 1;
569  break;
570  case (OTHER):
571  nset = 2;
572  break;
573  case (MARINERS_STANDARD):
574  nset = 3;
575  break;
576  default:
577  nset = 2;
578  break;
579  }
580 
581  if (m_pDispCat->GetSelection() != nset) {
582  int valSet = STANDARD;
583  int newSet = m_pDispCat->GetSelection();
584  switch (newSet) {
585  case 0:
586  valSet = DISPLAYBASE;
587  break;
588  case 1:
589  valSet = STANDARD;
590  break;
591  case 2:
592  valSet = OTHER;
593  break;
594  case 3:
595  valSet = MARINERS_STANDARD;
596  break;
597  default:
598  valSet = STANDARD;
599  break;
600  }
601  parentCanvas->SetENCDisplayCategory(valSet);
602  b_needReLoad = true;
603 
604  // Anchor conditions are only available if display category is "All" or
605  // "User Standard"
606  pCBENCAnchorDetails->Enable(newSet > 1);
607  }
608 
609  if (b_needReLoad) {
610  parentCanvas->ReloadVP();
611  } else if (b_needRefresh) {
612  parentCanvas->Refresh(true);
613  parentCanvas->InvalidateGL();
614  }
615 
616  RefreshControlValues();
617 }
