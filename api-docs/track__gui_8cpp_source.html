218 void TrackGui::Assemble(ChartCanvas *cc,
219  std::list<std::list<wxPoint> > &pointlists,
220  const LLBBox &box, double scale, int &last, int level,
221  int pos) {
222  if (pos == (int)m_track.SubTracks[level].size()) return;
223 
224  SubTrack &s = m_track.SubTracks[level][pos];
225  if (box.IntersectOut(s.m_box)) return;
226 
227  if (s.m_scale < scale) {
228  pos <<= level;
229 
230  if (last < pos - 1) {
231  std::list<wxPoint> new_list;
232  pointlists.push_back(new_list);
233  }
234 
235  if (last < pos) AddPointToList(cc, pointlists, pos);
236  last = wxMin(pos + (1 << level), m_track.TrackPoints.size() - 1);
237  AddPointToList(cc, pointlists, last);
238  } else {
239  Assemble(cc, pointlists, box, scale, last, level - 1, pos << 1);
240  Assemble(cc, pointlists, box, scale, last, level - 1, (pos << 1) + 1);
241  }
