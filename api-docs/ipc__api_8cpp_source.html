96 const void* IpcConnection::OnRequest(const wxString& topic,
97  const wxString& item, size_t* size,
98  wxIPCFormat format) {
99  if (format != wxIPC_TEXT) return 0;
100 
101  std::string line = item.ToStdString();
102  if (ocpn::startswith(line, "get_rest_endpoint")) {
103  buffer = server.get_rest_api_endpoint_cb();
104  if (size) *size = buffer.size();
105  return buffer.c_str();
106  } else if( ocpn::startswith(line, "open")) {
107  auto words = ocpn::split(line.c_str(), " ");
108  if (words.size() != 2) {
109  wxLogWarning("Illegal open cmd line: %s", line.c_str());
110  return 0;
111  }
112  bool ok = server.open_file_cb(words[1]);
113  const char* reply = ok ? "ok" : "fail";
114  if (size) *size = strlen(reply);
115  return reply;
116  } else {
117  wxLogWarning("Illegal cmd line: %s", line.c_str());
118  return 0;
119  }
120 }
