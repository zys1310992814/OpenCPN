97 void PluginPaths::initLinuxPaths() {
98  using namespace std;
99 
100  if (g_bportable) {
101  m_userLibdir = g_BasePlatform->GetPrivateDataDir().ToStdString() +
102  "/plugins/lib"; // m_home + "/.local/lib";
103  m_libdirs.push_back(m_userLibdir);
104  m_userBindir = g_BasePlatform->GetPrivateDataDir().ToStdString() +
105  "/plugins/bin"; // m_home + "/.local/bin";
106  m_bindirs = m_libdirs;
107  m_userDatadir = g_BasePlatform->GetPrivateDataDir().ToStdString() +
108  "/plugins/share"; // m_home + "/.local/share";
109  m_datadirs.push_back(m_userDatadir);
110  return;
111  }
112 
113  m_userLibdir = m_home + "/.local/lib";
114  m_userBindir = m_home + "/.local/bin";
115  m_userDatadir = m_home + "/.local/share";
116 
117  std::vector<std::string> base_plugin_paths;
118 #if defined(__WXGTK__) || defined(__WXQT__)
119  char exe_buf[100] = {0};
120  ssize_t len = readlink("/proc/self/exe", exe_buf, 99);
121  if (len > 0){
122  exe_buf[len] = '\0';
123  wxFileName fn(exe_buf);
124  std::string path = fn.GetPath().ToStdString();
125  base_plugin_paths.push_back(expand(path + "/../lib/opencpn"));
126  if(g_BasePlatform->GetOSDetail()->osd_arch.find("64") != string::npos) {
127  base_plugin_paths.push_back(expand(path + "/../lib64/opencpn"));
128  } else {
129  base_plugin_paths.push_back(expand(path + "/../lib32/opencpn"));
130  }
131  }
132 #endif
133 
134  const char* const envdirs = getenv("OPENCPN_PLUGIN_DIRS");
135  string dirlist = envdirs ? envdirs : "~/.local/lib/opencpn";
136  m_libdirs = split(dirlist, ':');
137  for (auto& dir : m_libdirs) {
138  dir = expand(dir);
139  }
140  for (auto &base_plugin_path : base_plugin_paths) {
141  if (envdirs == 0 && dirlist.find(base_plugin_path) == string::npos) {
142  if(ocpn::exists(base_plugin_path)) {
143  m_libdirs.push_back(base_plugin_path);
144  }
145  }
146  }
147  m_bindirs = m_libdirs;
148  for (auto& dir : m_bindirs) {
149  // Fails on Debian multilib paths like /usr/lib/x86_64-linux-gnu.
150  // But we don't use those even on Debian.
151  size_t pos = dir.rfind("/lib/opencpn");
152  if (pos == string::npos) {
153  pos = dir.rfind("/lib64/opencpn");
154  }
155  dir = pos == string::npos ? dir : dir.substr(0, pos) + "/bin";
156  }
157  const char* const xdg_data_dirs = getenv("XDG_DATA_DIRS");
158  dirlist = xdg_data_dirs ? xdg_data_dirs : "~/.local/lib";
159  m_datadirs = split(dirlist, ':');
160  for (auto& dir : m_datadirs) {
161  dir += "/opencpn/plugins";
162  }
163  for (auto &base_plugin_path : base_plugin_paths) {
164  if (xdg_data_dirs == 0 && dirlist.find(base_plugin_path) == string::npos) {
165  m_datadirs.push_back(base_plugin_path + "/plugins");
166  }
167  }
168 }
